\section{Liquid Lasanga}


\begin{recipestats}[
	servings=4,
	preptime=20 \minute,
	bakingtime=20 \minute,
	source=HGTV Magazine,
]
\end{recipestats}


\begin{recipeabstract}
	American lasagna in liquid form.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[8][\ounce]{lasgna noodles}\\
		\ingredient[\onehalf][\pound]{Italian sausage}\\
		\ingredient[1]{onion, white}\\
		\ingredient[3][cloves]{garlic}\\
		\ingredient[1][\teaspoon]{oregano}\\
		\ingredient[2][\Tablespoon]{tomato paste}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[4][C]{broth, chicken}\\
		\ingredient[15][\ounce]{tomatoes, crushed}\\
		\ingredient[\onehalf][C]{basil}\\
		\ingredient[\onehalf][C]{parmesan}\\
		\ingredient[\onefourth][C]{cream}
	\end{ingredientblock}
\end{ingredientcolumns}


\ragmarpar{hot Italian sausage is best}
\begin{preparation}
\item Reserve a $\geq 3.5 \quart$ pot for both the soup and noodles.
	Break the noodles into pieces.
	Chop the onion.
	Smash and peel the garlic.

\item Cook the noodles and toss with oil.

\item Meanwhile brown the sausage on medium high, brake apart as you go.
	Add the onion and saute\`{e} $[2 \ldots 4]$ \minute.
	Add the garlic, oregano, tomato paste, cook $[2 \ldots 4]$ \minute.
\end{preparation}
