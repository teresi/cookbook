\section[Turkey Gumbo]{Thanksgiving Turkey Gumbo}\index{turkey!Turkey Gumbo}


\begin{recipestats}[
	servings=[4 \ldots 6] people,
	preptime=30~\minute,
	bakingtime=4~\hour,
	source=Teresi Family,
]
\end{recipestats}


\begin{recipeabstract}
	A Teresi tradition; Scott says it is ``the best part of Thanksgiving.''
	Create a stock with the turkey bones for some extra pizazz.
\end{recipeabstract}


\ragmarpar{Andoullie sausage is best.}
\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[12][\ounce]{smoked sausage}\\
		\ingredient[2][C]{shredded turkey}\\
		\ingredient[\nicefrac{1}{4}][C]{vegetable oil}\\
		\ingredient[7][\Tablespoon]{flour}\\
		\ingredient[2][C]{white onions}\\
		\ingredient[\nicefrac{3}{2}][C]{celery}\\
		\ingredient[\nicefrac{1}{2}][C]{green bell pepper}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[4][C]{stock}\\
		\ingredient[1][tsp]{gumbo fil\'{e}}\\
		\ingredient[1][tsp]{salt}\\
		\ingredient[1][tsp]{marjaoram}\\
		\ingredient[1][tsp]{white pepper}\\
		\ingredient[\approx~1][C]{white rice}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Reserve a dutch oven and skillet.

\item Dice vegetables \& turkey, rangiri cut the sausage, set aside.

\item Make the roux.
	Combine flour \& oil, stirring constantly on medium-high heat until light brown, $\approx 15 \ldots 20$ \minute.

\item Add celery, onions, salt; cook $\approx 10$ \minute; deglaze w/ beer.

\item Meanwhile lightly brown the sausage.

\item Add turkey, sausage, stock. Simmer $\approx 4$ \hour.

\item Prepare rice to serve when gumbo is complete.
\ragmarpar{Serve w/ Ciabatta.}

\item Add gumbo fil\'{e} \& marjoram, then salt \& pepper to taste.
\end{preparation}


\begin{experiments}

\end{experiments}
