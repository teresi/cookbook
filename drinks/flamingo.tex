\section[Flamingo]{Flamingo~\vegan}
\index{pineapple!Flamingo}


\begin{recipestats}[
	servings=1,
	preptime=5 \minute,
	source=Mike \& Jane,
	original=\citefield{bonnevilleCocktailCollectionRumIII}{shorttitle}\cite{bonnevilleCocktailCollectionRumIII},
]
\end{recipestats}


\begin{recipeabstract}
	Rum, pineapple, pink.
	The pineapple provides a nice foam top when shaken for an impressive display in a coupe glass.
\end{recipeabstract}


\ragmarpar{Shake pineapple juice first as the sediment improves the taste.}
\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[1 \onehalf][\ounce]{rum}\\
		\ingredient[1][\ounce]{pineapple juice}\\
		\ingredient[\onehalf][\ounce]{lime juice}\\
		\ingredient[1 \onehalf][\teaspoon]{grenadine}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Add ice to shaker, add ingredients, shake and serve in a coupe.
\end{preparation}


\recipeend
