\section[Pacific Sushine]{Pacific Sunshine\ragStars{3}}


\begin{recipestats}[
	servings=1,
	preptime=5 \minute,
	source=Mike \& Jane,
	original=\citefield{mrBoston2006}{shorttitle} \cite{mrBoston2006},
]
\end{recipestats}


\begin{recipeabstract}
	As blue as the Pacific.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1 \onehalf][\fluidounce]{tequila}\\
		\ingredient[1 \onehalf][\fluidounce]{blue cura\c{c}ao}\\
		\ingredient[\threefourth][\fluidounce]{simple syrup}
	\end{ingredientblock}

	\begin{ingredientblock}
		\ingredient[\threefourth][\fluidounce]{lemon juice}\\
		\ingredient[2][dash]{Angostura bitters}\\
		\ingredient[1]{Maraschino cherry}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Shake with ice.

\item Pour all into chilled tulip glass.
	Fill with crushed ice.

\item Garnish with the cherry and a paper umbrella.
\end{preparation}


\recipeend
