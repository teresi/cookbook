\section[Rich Syrup]{Rich Syrup}\label{richSyrup}


\begin{recipestats}[
	servings=$\approx~1~Cup$,
	preptime=5~\minute,
	original=\citefield{cocktailSeminars2021}{title}~\cite{cocktailSeminars2021},
]
\end{recipestats}

\ragmarpar{Rich syrup dilutes a drink less than \nameref{simpleSyrup} \& provides a silky mouthfeel.}
\begin{recipeabstract}
	Rich syrup is a 2:1 by volume mix of water and sucrose.
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[\onethird][C]{water}\\
		\ingredient[\twothird][C]{white sugar}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Microwave water to near boil, combine with sugar.
\item Bottle, cool, and refridgerate up to $3~weeks$.
\end{preparation}


\recipeend
