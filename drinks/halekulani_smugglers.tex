\begin{recipeabstract}
	A more balanced \nameref{halekulani} from \citeauthor{smugglersCove}, achieved with the addition of pineapple juice and the partial replacement of grenadine with Demerara syrup.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1\onehalf][\fluidounce]{bourbon}\\
		\ingredient[\onehalf][\fluidounce]{lemon juice}\\
		\ingredient[\onehalf][\fluidounce]{orange juice}\\
		\ingredient[\onehalf][\fluidounce]{pineapple juice}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\onefourth][\fluidounce]{demerara syrup}\\
		\ingredient[\onehalf][\teaspoon]{grenadine}\\
		\ingredient[1][dash]{Angostura}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Shake with ice.
\item Strain into a chilled coupe, garnish with an edible orchid.
\end{preparation}

\recipeend
