\section[Mexican El Diablo]{Mexican El Diablo~\ragStars{2}}


\begin{recipestats}[
	servings=1,
	preptime=5~\minute,
	source=\citeauthor{smugglersCove}~\cite[p.~43]{smugglersCove},
	original=\citeauthor{traderVicsFoodAndDrink}~\cite{traderVicsFoodAndDrink},
]

\end{recipestats}

\begin{recipeabstract}
	A quick drink.
	The ginger ale works well with the cassis.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1][wedge]{lime}\\
		\ingredient[\onehalf][\fluidounce]{lime juice}\\
		\ingredient[\onehalf][\fluidounce]{cr\'{e}me de cassis}\\
	\end{ingredientblock}

	\begin{ingredientblock}
		\ingredient[1\onehalf][\fluidounce]{white tequila}\\
		\ingredient[4][\fluidounce]{ginger ale}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
	\item Squeeze lime wedge into collins glass, add to glass.
	\item Fill glass with crushed ice.
	\item Add remaining to glass, stir
\end{preparation}


\recipeend%
