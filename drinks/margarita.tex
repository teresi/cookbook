\section[Original Margarita]{Original Margarita\ragStars{3}}\index{lime!Original Margarita}\index{tequila!Original Margarita}


\begin{recipestats}[
	servings=1,
	preptime=5 \minute,
	source=\citefield{cointreauMargarita}{shorttitle}~\cite{cointreauMargarita},
]
\end{recipestats}


\ragmarpar{Mike likes this.}
\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[2][\fluidounce]{tequila, white}\\
		\ingredient[1][\fluidounce]{Cointreau}\\
		\ingredient[1][\fluidounce]{lime juice}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Shake ingredients with ice.
\item Pour into highball with ice.
\end{preparation}
\ragmarpar{Add a dash of Agave or honey.}


\begin{variation}
\item Substitue Cointreau with Triple Sec.
\end{variation}


\recipeend%
