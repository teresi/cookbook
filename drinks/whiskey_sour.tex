\ragmarpar{Jane prefers the sour with a cherry.}
\begin{recipeabstract}
	An early classic, named a ``sour'' as it is spirit with sugar and citrus juice.
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[2][\fluidounce]{whiskey}\\
		\ingredient[1][\fluidounce]{simple syrup}\\
		\ingredient[\threefourth][\fluidounce]{lemon juice}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Shake with ice and pour into a coupe or rocks glass.
\item Optionally garnish with a lemon twist, and/or a block of ice.
\end{preparation}


\begin{variation}
\item Replace syrup with~\ref{gingerSyrup}~\nameref{gingerSyrup} for extra pizazz.
\item Freeze some cranberries into a large ice block for a holiday theme.
\end{variation}


\recipeend%
