\begin{recipeabstract}
	A whiskey sour with part of the syrup replaced by cr\'{e}me de cassis, and a little extra lemon juice to round it out.
	Cr\'{e}me de cassis is a black current liquer with a rich and tart berry flavor.
	Jeffrey thinks the liquer works well with the lemon\cite{bourbonRenewalMorgenthaler}.
\end{recipeabstract}


\ragmarpar{Mike like cr\'{e}me de cassis for it's unique flavor.}
\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[2][\fluidounce]{Bourbon}\\
		\ingredient[1][\fluidounce]{lemon juice}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\onehalf][\fluidounce]{cr\'{e}me de cassis}\\
		\ingredient[\onehalf][\fluidounce]{simple syrup}\\
		\ingredient[1][dash]{Angostura}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
	\item Shake with ice, pour over ice in a rocks glass.
	\item Garnish w/ lemon wedge or currants.
\end{preparation}


\recipeend%
