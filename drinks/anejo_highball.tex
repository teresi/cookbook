\begin{recipeabstract}
	Similar to a Dark and Stormy but not as spicy and more balanced.
\end{recipeabstract}


\ragmarpar{Jane likes this.}
\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1~\onehalf][\fluidounce]{aged rum}\\
		\ingredient[\onehalf][\fluidounce]{cura\c{c}ao}\\
		\ingredient[\onefourth][\fluidounce]{lime juice}
	\end{ingredientblock}

	\begin{ingredientblock}
		\ingredient[2][dash]{Agostura}\\
		\ingredient[2][\fluidounce]{ginger beer}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
	\item Shake all but the soda with ice.
		Fill a highball w/ crushed ice.
	\item Strain into glass, top with ginger beer.
	\item Garnish with lime and/or orange slice.
\end{preparation}
