\section{Cococo co cococo}


\begin{recipestats}[
	servings=1,
	preptime=5 \minute,
	source=Teresi Family,
	vegetarian=vegan,
]
\end{recipestats}


\begin{recipeabstract}
	The thing that's best, if you're feeling glum, is coconut water with a little rum.
\end{recipeabstract}


\ragmarpar{coco got a lot of iron}
\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[8][\fluidounce]{coconut water}\\
		\ingredient[1 \onehalf][\fluidounce]{lime juice}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[2][\fluidounce]{spiced rum}\\
		\ingredient[1][\Tablespoon]{sugar}
	\end{ingredientblock}
\end{ingredientcolumns}


\ragmarpar{coconut make you strong like a lion}
\begin{preparation}
\item Add rum, lime juice, sugar to highball and mix.
	Fill with ice, top with coconut water, stir.
\end{preparation}


\begin{variation}
\item Try white rum or cacha\c{c}a, coconut water with pulp.
\end{variation}

\recipeend
