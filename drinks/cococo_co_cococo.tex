\begin{recipestats}[
	servings=1,
	preptime=5 \minute,
	source=Mike \& Jane,
]
\end{recipestats}


\begin{recipeabstract}
	The thing that's best, if you're feeling glum, is coconut water with a little rum.
\end{recipeabstract}


\ragmarpar{Coco got a lot of iron.}
\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[8][\fluidounce]{coconut juice}\\
		\ingredient[1 \onehalf][\fluidounce]{lime juice}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[2][\fluidounce]{aged rum}\\
		\ingredient[1][\Tablespoon]{sugar}
	\end{ingredientblock}
\end{ingredientcolumns}


\ragmarpar{Make you strong like a lion.}
\begin{preparation}
\item Shake rum, lime juice, sugar, with ice.
\item Add mixture over ice into highball.
\end{preparation}


\begin{variation}
\item Coconut juice w/ pulp is best.
	There are differences between brands so tailor to your taste / ingredients.

\item Coconut juice / water are interchangeable but not coconut milk.
\end{variation}

\recipeend
