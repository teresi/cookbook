\section[Pegu Club]{Pegu Club~\vegan}
\index{gin!Pegu Club}

\begin{recipestats}[
	servings=1,
	preptime=5~\minute,
	original=\citefield{cocktailSeminars2021}{title} \cite{cocktailSeminars2021},
]
\end{recipestats}


\begin{recipeabstract}
	A classic drink.
	The bitters help to counteract the sour citrus.
	Recommended for those who may not normally like gin as the bitters give it a pleasant depth of spice.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1 \onehalf][\fluidounce]{gin, dry}\\
		\ingredient[\threefourth][\fluidounce]{triple sec}\\
		\ingredient[\threefourth][\fluidounce]{lime juice}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[1][dash]{Angostura bitters}\\
		\ingredient[1][dash]{orange bitters}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Shake with ice, pour into coupe.
\item Optionally garnish with lime peel.
\end{preparation}


\begin{variation}
\item Replace orange bitters with Angostura
\end{variation}


\recipeend
