\begin{recipeabstract}
	Simple syrup is a 1:1 by volume mix of water and sucrose.
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[\onehalf][C]{water}\\
		\ingredient[\onehalf][C]{white sugar}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Microwave water to near boil, combine with sugar.
\item Bottle, cool, and refridgerate up to $3~weeks$.
\end{preparation}
