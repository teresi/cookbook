\begin{recipeabstract}
	An unusual Tiki drink for it's use of bourbon.
	Courtesy of the Halekulani Hotel all the way from 1932 Waikiki, Hawaii.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1\onehalf][\fluidounce]{bourbon}\\
		\ingredient[\onehalf][\fluidounce]{lemon juice}\\
		\ingredient[\onehalf][\fluidounce]{orange juice}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\onehalf][\fluidounce]{grenadine}\\
		\ingredient[1][dash]{Angostura}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Shake with ice.
\item Strain into a coupe, optionally garnish w/ citrus of your choice.
\end{preparation}

\begin{experiments}
	\item Looking to try custom grenadine given it's high volume, as it sometimes can overpower the other ingredients.
\end{experiments}
