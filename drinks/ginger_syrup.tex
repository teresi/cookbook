\section{Ginger Simple Syrup}

\begin{recipestats}[
	servings=$\approx~1~Cup$,
	preptime=5~\minute,
	original=\citefield{alohaKitchen}{shorttitle}~\cite{alohaKitchen},
]
\end{recipestats}

\begin{recipeabstract}
	Ginger flavored simple syrup.
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[\onehalf][C]{water}\\
		\ingredient[\onehalf][C]{white sugar}\\
		\ingredient[\approx~1][inch]{ginger root}%
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
	\item Peel and crush ginger into saucepan, add water \& sugar.
	\item Bring to simmer and stir over medium heat until filly dissolved.
	\item Take off heat, steep $\approx 1 \hour$, strain \& bottle.
\end{preparation}


\recipeend%
