\section{Overnight Oats}


\begin{recipestats}[
	servings=3,
	preptime=10 \minute,
	inactivetime=12 \hour,
	source=Mike \& Jane,
	original=\citetitle{kwonOvernightOats} \cite{kwonOvernightOats},
]
\end{recipestats}


\begin{recipeabstract}
	A quick breakfast to make ahead on the weekend.
\end{recipeabstract}

\ragmarpar{a just green banana is best}
\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[\onehalf][C]{oats (quick $\|$ rolled)}\\
		\ingredient[\onehalf][C]{milk}\\
		\ingredient[\onehalf][C]{Greek yogurt}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[1]{banana}\\
		\ingredient[\approx 4][\Tablespoon]{strawberry jam}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Mix oats, milk, yogurt, jam.
	Slice banana, add.

\item Refrigerate overnight.
	Use within 3 days.
\end{preparation}


\begin{variation}
\item Other jam and fruits are available.
	We recommend Blackberry.

\ragmarpar{See \cite{kwonOvernightOats} for many variations}
\item Peanut butter, granola, chia seeds, etc. are popular.
\end{variation}


\recipeend
