\begin{recipeabstract}
	A Teresi Christmas tradition.
	Prepare the night before for an easy breakfast.
	See \ref{sausage_casserole_christmas} \nameref{sausage_casserole_christmas} for the savory counterpart.
\end{recipeabstract}

\ragmarpar{An italian, challah, or brioche loaf all work.}
\begin{preparation}
\item Grease a $\approx13x9$ $\inch$ pan.

\item Mix cinnamon and brown sugar, set aside.

\item Dice bread loaf into $\approx[1..2]$ $\inch$ cubes, add to pan.

\item Beat eggs in a large bowl, beat in cream, fold mixture into bread.

\item Cover pan and refridgerate overnight.

\item Next morning preheat oven to 350 $\Fahrenheit$.

\item Sprinkle sugar on top of casserole, melt butter, pour over sugar.

\item Bake 35 $\minute$, let cool 10 $\minute$.
	Grate some nutmeg on top, serve w/ maple syrup.
\end{preparation}


\begin{variation}
\item Try also with pecans or mixed berries.
\end{variation}
