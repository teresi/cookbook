\begin{recipe}[
    preparationtime = 20 minutes,
    bakingtime = 30 minutes,
    portion = 2...4 people,
    source = Blue Apron
]
{Mexican Beef Bowl}
\graph{
    %small = strawberry,
    smallpicturewidth = 0.3\textwidth,
    %big = strawberrycake,
    bigpicturewidth = 0.6\textwidth,
}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\ingredients{
Beef:\\\\
\unit[1]{lb} & ground beef\\
\unit[1/2]{lb} & tomatoes, cherry\\
\unit[1/2]{lb} & carrots\\
\unit[1]{} & lime\\
\unit[1/4]{C} & jalepe\~{n}os, pickled\\
\unit[1/4]{C} & sour cream\\
\unit[4...8]{oz} & cheese, cotija\\
\unit[3]{Tbs} & spice blend\\
\unit[1]{C} & rice\\
\\
Spices:\\\\
\unit[2]{Tbs} & Guajillo powder\\
\unit[1]{Tbs} & chili powder\\
\unit[1]{Tbs} & paprika, smoked\\
\unit[1]{Tbs} & cumin\\
\unit[1]{Tbs} & marjoram\\
\unit[1/2]{Tbs} & garlic, ground\\
\unit[1/2]{Tbs} & salt
}

\label{mexican_beef_bowl}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\preparation{

\step Begin to brown the beef in large saut\'{e} pan, stirring occasionally. Reach a strong color at end of recipe.

\step Dice carrots, toss with oil and spices.\\Preheat oven to $450 F^\circ$.

\step Begin cooking rice.

\step Roast carrots in oven on a cookie sheet, $12...14\;min$.

\step Dice tomatoes, finely dice jalepe\~{n}os, combine with juice of 1/2 of lime, pickled jalepe\~{n}o juice to taste.

\step Add spices to beef ($\approx 3/4\; Tbs$) at end, cook $\approx1\; min$.\\Add $1/4\;C$ water, cook $2...3\;min$.

\step Combine mayonnaise and juice of 1/2 lime.\\Add pickled jalepe\~{n}o juice to taste.

\step Serve by layering rice, beef, vegetables, pickled jalepe\~{n}os, cheese, mayonnaise.

\vspace{-2cm}
}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\hint{
\begin{itemize}
\item Chorizo works well in place of part or all of the beef. Season accordingly.
\item Vary the amount of tomotoes, carrots, jalepe\~{n}os to taste. More carrots works particularly well.
\item Cotija cheese is worth the effort to find it, but melted mozarella works too.
\item Try replacing the rice with riced cauliflower. Cut into small pieces, chop in food processor with garlic, optionally frying in pan with or without oil.
\item Ancho chili powder words particularly well.
\item For the Guajillo powder, buy the dried chilies, remove the seeds, grind in a coffee grinder.
\item The spice list is more of a guideline, experiment with the recipe, such as adding a few Arbol or another dried chili.
\end{itemize}
}
