\begin{recipeabstract}
	Chicken with a sweet soy sauce and a bit of ginger.
	It is quite simple and can be made ahead.
	Goes well with rice, bread, or something that can soak up the sauce.
\end{recipeabstract}


\ragmarpar{$\approx2~\Tablespoon$ minced garlic.}
\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[2~\onehalf][\pound]{chicken thighs}\\
		\ingredient[\onethird][C]{honey}\\
		\ingredient[\onethird][C]{soy sauce}\\
		\ingredient[\onefourth][C]{green onion}\\
		\ingredient[2][\Tablespoon]{oil}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[1][\Tablespoon]{vinegar}\\
		\ingredient[6][cloves]{garlic}\\
		\ingredient[1][\teaspoon]{sesame oil}\\
		\ingredient[1][\teaspoon]{ginger, crushed}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Chop green onion, smash and peel garlic, peel ginger.

\item Mix honey, soy sauce, green onion, oil, vinegar, and sesame oil.
	Crush garlic and ginger into mix.
	Stir to dissolve.

\item Combine chicken, marinade.
	Marinate $\ge30$ \minute, or overnight.

\item Preheat oven to $425$ \Fahrenheit.
	Transfer chicken and marinade to a cast iron pan.

\item Bake $20 \dots 25$ \minute, flipping halfway through.
	Broil $\approx5$ \minute.

\item Remove and serve.
	Optionally garnish with sesame seeds and/or parsely.

\end{preparation}
