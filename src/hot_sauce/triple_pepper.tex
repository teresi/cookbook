\section[Triple Pepper]{Triple Pepper Hot Sauce}


\begin{recipestats}[
	servings=16~\fluidounce,
	preptime=45~\minute,
	bakingtime=30~\minute,
	inactivetime=2~\week,
	original=\citefield{newCookBook2014}{shorttitle}~\cite[p.~201]{newCookBook2014},
]
\end{recipestats}


\begin{recipeabstract}
	A surprisingly mild hot sauce.
	An orange oaisis.
	Perfect for the Teresi Christmas breakfast casserole.
\end{recipeabstract}


\ragmarpar{Use 4 or 5 Habenero.}
\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[2][C]{vinegar, white}\\
		\ingredient[1\onehalf][C]{carrot}\\
		\ingredient[1][C]{onion, white}\\
		\ingredient[\onefourth][C]{lime juice}\\
		\ingredient[3][clove]{garlic}\\
		\ingredient[1][\teaspoon]{salt}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\onehalf][C]{sweet pepper}\\
		\ingredient[$[3..6]$][]{Habenero chile}\\
		\ingredient[3][]{Fresno chile}\\
		\ingredient[3][]{Thai chile}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Reserve four 4 $\fluidounce$ canning jars, sauce pan, soup pot.
\item Sterilize jars and lids.
\ragmarpar{Wear gloves!}
\item Finely chop all the vegetables, crush the garlic.
\item Combine in a large sauce pan all except the peppers.
\item Bring sauce to boil, reduce heat, simmer uncovered $[20 .. 25]$ $\minute$.
\ragmarpar{Use bottled lime juice; it's acidity is more predictable.}
\item Remove from heat, leat cool.
\item Transfer to a blender, add peppers, blend until smooth.
\item Strain through a fine seive, return to sauce pan.
\item Bring to boil over medium-high, reduce to low, simmer 10 $\minute$ uncovered.
\item Add sauce into the jars, leave $\onefourth$ $\inch$ headspace, wipe jar rims and lightly tighten lids.
\item Move jars to boiling water canner and bring to boil.
	Boil for $5$ $\minute$.
\item Remove and let cool, label jars with the canning date.
\item Let stand at room temperature 2 $\week$ before serving.
\end{preparation}


\recipeend%
