\begin{recipeabstract}
	A Teresi Christmas tradition.
	Prepare the night before for an easy breakfast.
	See~\ref{french_toast_casserole_christmas}~\nameref{french_toast_casserole_christmas} for the sweet counterpart.
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[1][\pound]{breakfast sausage}\\
		\ingredient[\onehalf][\pound]{bacon}\\
		\ingredient[10][\ounce]{croutons}\\
		\ingredient[4][large]{eggs}\\
        \ingredient[10.5][\ounce]{cream of mushroom soup}\\
		\ingredient[8][\ounce]{mushrooms, canned}\\
		\ingredient[2 \onefourth][C]{whole milk}\\
		\ingredient[\threefourth][\teaspoon]{dry mustard}\\
		\ingredient[\threefourth][\pound]{cheddar}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Reserve a large skillet, and $13\times9''$ pan.

\item Slice bacon to $\approx2$ $\inch$ strips, fry in skillet.

\item Meanwhile grease the pan and add the croutons.

\item Remove bacon just before it starts to get crispy.

\ragmarpar{Use 16 $\ounce$ fresh mushrooms if not canned, cook with bacon.}
\item Start cooking sausage on medium high. Drain mushrooms.

\item Beat eggs in a large bowl.\\
    Add milk, mustard, cream of mushroom soup.

\item Let meat cool, cover croutons with bacon \& sausage.

\item Covert croutons w/ the mushrooms, bacon, sausage.

\item Pour in the egg mixture.

\item Seal with lid and refridgerate overnight.

\item In the morning remove from fridge and let stand for 30 $\minute$.

\item Preheat oven to 350 $\Fahrenheit$.

\ragmarpar{Serve with Tabasco or Cholula.}
\item Bake for 50 $\minute$.
	Meanwhile, shred the cheese, set aside.

\item Remove, top with cheddar, then bake another 5 $\minute$.
\end{preparation}
