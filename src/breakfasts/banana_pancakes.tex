\section{Banana Pancakes}\index{banana!Banana~Pancake}


\begin{recipestats}[
	servings=12 pancakes,
	preptime=10 \minute,
	source=Mike \& Jane,
]
\end{recipestats}

\ragmarpar{Keep your ripe bananas in the freezer and defrost prior.}
\begin{recipeabstract}
	A breakfast to make on the weekend after sleeping in.
	Save extra pancakes in foil on a single layer to bake later.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1][C]{flour, all-purpose}\\
		\ingredient[2][\Tablespoon]{sugar}\\
		\ingredient[2][\teaspoon]{baking powder}\\
		\ingredient[\onehalf][\teaspoon]{salt}
	\end{ingredientblock}

	\begin{ingredientblock}
		\ingredient[\onehalf][C]{Greek yogurt}\\
		\ingredient[\onehalf][C]{milk, whole}\\
		\ingredient[\approx7..9][\ounce]{banana, ripe}\\
		\ingredient[1]{egg}\\
	\end{ingredientblock}
\end{ingredientcolumns}
\ragmarpar{About 2 bananas}


\ragmarpar{Add 1 \teaspoon vanilla.}
\begin{preparation}
\item Mix flour, sugar, baking powder, salt, set aside.

\item Begin to preheat your pan on medium low to medium.

\item Peal bananas into a large bowl, mash briefly with fork.

\item Mix in yogurt, egg, milk, beat with fork.

\item Fold in the flour to the banana mixture until just combined.

\item Add a small amount of butter or oil to your skillet, then two spoonfuls of batter per each cake.
	Cook on first side until bubbles come through.
	Flip and cook until lightly brown, $\approx2$ \minute.
\ragmarpar{An iron skillet is critical.}

\item Adjust heat after each batch if necessary, repeat.

\item Serve w/ syrup and apricot or strawberry jam.
\end{preparation}


\recipeend%
