\section{Overnight Oats}\index{oats!Overnight~Oats}


\begin{recipestats}[
	servings=3,
	preptime=10 \minute,
	inactivetime=12 \hour,
	source=Mike \& Jane,
	original=\citetitle{kwonOvernightOats}~\cite{kwonOvernightOats},
]
\end{recipestats}


\begin{recipeabstract}
	A quick breakfast to make ahead on the weekend.
\end{recipeabstract}

\ragmarpar{A just green banana is best.}
\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[\onehalf][C]{oats (quick $\|$ rolled)}\\
		\ingredient[\onehalf][C]{milk}\\
		\ingredient[\onehalf][C]{Greek yogurt}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[1]{banana}\\
		\ingredient[\approx~4][\Tablespoon]{strawberry jam}
	\end{ingredientblock}
\end{ingredientcolumns}
\ragmarpar{Rolled oats will keep their shape moreso than quick.}


\begin{preparation}
\item Mix oats, milk, yogurt, jam, and sliced banana.

\item Refrigerate overnight.
	Use within 3 days.
\end{preparation}


\begin{variation}
\item Jane adds honey, prefers strawberries to banana.

\ragmarpar{See~\cite{kwonOvernightOats} for variations.}
\item Other jam, peanut butter, granola, etc.\ are popular.
\end{variation}


\recipeend%
