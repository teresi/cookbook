\section{Egg Muffins}\index{egg!Egg Muffins}


\begin{recipestats}[
	servings=10,
	preptime=30 \minute,
	bakingtime=25 \minute,
	source=Mike \& Jane,
]
\end{recipestats}


\begin{recipeabstract}
	A quick breakfast to make ahead on the weekend.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[8]{eggs}\\
		\ingredient[\onefourth][C]{milk}\\
		\ingredient[1]{onion, red}\\
		\ingredient[4][slices]{bacon}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[1][small]{potato, waxy}\\
		\ingredient[4][\ounce]{monterey jack}\\
		\ingredient[1][\teaspoon]{salt}\\
		\ingredient[1][\teaspoon]{marjoram}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Begin cooking bacon on medium in a skillet, flipping occasionaly.
	Remove when cooked through and chop.

\ragmarpar{Cut onion in half along axis then slice along axis, so it cooks evenly.}
\item Meanwhile, lyonnaise cut the onion, dice cheese and potato.

\item Whisk in a large bowl the eggs, milk, salt, marjoram.

\item Add cupcake liners to muffin tin.
	Preheat oven to 350 \Fahrenheit.

\item Add potato to skillet, stirring occasionally.
	Cook $\approx 5$ \minute.

\item Add onion to skillet, cook $\approx [5..10]$ \minute.

\item Remove from heat when onion is done.
	Spoon out vegetables into a small bowl to strain, mix in bacon.

\item Pour eggs into muffin tins about \threefourth~full.
	Add vegetable mix and cheese on top.
	Add water to unfilled tins.
	Bake 25 \minute.
\end{preparation}


\recipeend%
