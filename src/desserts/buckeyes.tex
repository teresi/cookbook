\begin{recipestats}[
	servings=48,
	preptime=15~\minute,
	inactivetime=30~\minute,
	original=Jim Giesken,
]
\end{recipestats}


\ragmarpar{Jim likes milk chocolate.}
\begin{recipeabstract}
    A Giesken and The Ohio State Christmas tradition.
    Peanut butter is dipped in chocolate to resemble a Buckeye, the nickname for Ohio State students and a poisonous nut.
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[16.3][\ounce]{peanut butter}\\
        \ingredient[12][\ounce]{confectioner's sugar}\\
        \ingredient[21][\ounce]{chocolate}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\ragmarpar{Take your rings off first.}
\item Reserve two cookie sheets, line w/ wax paper.
\item Mix the peanut butter and sugar with your hands.
\item Roll one ball of peanut butter as a test.
\item If the ball is too oily, add more sugar and remix.\\
    The sheen will diminish when you add sugar.
\ragmarpar{The peanut butter balls are $[14..18]~\gram$ each.}
\item Roll the peanut butter into 1 $\inch$ diameter balls.
\item Freeze for 30 $\minute$.
\item Melt the chocolate.
\item Use a fondue fork to partially dip into the chocolate, leaving the top third open.
    Move back onto the sheet and let solidify.
\item Press the top peanut butter to close the holes.
\end{preparation}


\recipeend%
