\section[Quick Pickles]{Basic Quick Pickles}\label{quick_pickles}\label{basic_quick_pickles}


\begin{recipestats}[
	servings=16 \fluidounce,
	preptime=30 \minute,
	bakingtime=10 \minute,
	source=Teresi Family,
]
\end{recipestats}


\begin{recipeabstract}
	These pickles are ``quick'' since they don't require canning.
	They are stored in the fridge as a result.
	We like to make jalepe\~{n}os and red onions, both with plenty of garlic.
\end{recipeabstract}


\ragmarpar{Prefer un-iodized salt to prevent discoloration.}
\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1][16 \ounce]{canning jar}\\
		\ingredient[\threefourth][C]{vinegar}\\
		\ingredient[\onehalf][C]{water}\\
		\ingredient[1][]{onion, red}\\
		\ingredient[1][]{jalepe\~{n}o}\\
		\ingredient[2][]{carrots}\\
		\ingredient[2][]{radishes}
	\end{ingredientblock}

	\bigskip
	\begin{ingredientblock}
		\ingredient[\onehalf][\Tablespoon]{sugar}\\
		\ingredient[\onehalf][\Tablespoon]{salt, kosher}\\
		\ingredient[2][cloves]{garlic}\\
		\ingredient[\onefourth][\teaspoon]{oregano}\\
		\ingredient[\onefourth][\teaspoon]{cumin seeds}\\
		\ingredient[\onefourth][\teaspoon]{mustard seeds}\\
		\ingredient[1][]{bay leaf}
	\end{ingredientblock}
\end{ingredientcolumns}


\ragmarpar{Pink peppercorns or coriander not recommended.}
\begin{preparation}
\item Add spices, salt, sugar, to the jar.
\item Slice the garlic and add.
\item Slice vegetables and pack into the jar.
\item Bring vinegar and water to a boil.
\item Pour over the vegetables, leave some space, tighten lid.
\item Let cool on the counter.
\item Move to fridge.
\end{preparation}


\recipeend%
