\section[Montreal Steak]{Montreal Steak Seasoning}\index{Rub!Montreal~Steak~Seasoning}


\begin{recipestats}[
	servings=5\onehalf~\Tablespoon,
	preptime=5~\minute,
	original=\citeauthor{spruceMontrealSteak}~\cite{spruceMontrealSteak},
]
\end{recipestats}

\begin{recipeabstract}
	A classic rub hailing from the delis of Canada and born from pickling spices.
	Slightly spicy with a strong but not overpowering coriander flavor.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1][\Tablespoon]{peppercorn, black}\\
		\ingredient[1][\Tablespoon]{paprika}\\
		\ingredient[1][\Tablespoon]{salt, kosher}\\
		\ingredient[\onehalf][\Tablespoon]{garlic, granulated}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\onehalf][\Tablespoon]{onion, granulated}\\
		\ingredient[\onehalf][\Tablespoon]{cayenne}\\
		\ingredient[\onehalf][\Tablespoon]{coriander}\\
		\ingredient[\onehalf][\Tablespoon]{dill seeds}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Coarse grind all but the granulated onion and garlic, combine all.
\item Add to meat either immediately before cooking or dry brine $\ge1$ $\hour$.
\end{preparation}

\begin{variation}
\item Make into a marinade by adding $\onehalf$ $\teaspoon$~of mix to $\onefourth~C$~oil, $\onefourth~C$~water, $2~\Tablespoon$~vinegar.
\end{variation}



\recipeend%
