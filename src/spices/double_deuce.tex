\section[Double Deuce]{Double Deuce Chicken Rub\ragStars{3}}\label{double_deuce_rub}\index{BBQ!Double~Deuce}


\begin{recipestats}[
	servings=1~\pound,
	preptime=10~\minute,
	original=Kevin \& Mike,
]
\end{recipestats}
\ragmarpar{Grill boneless chicken thighs for tacos, or BBQ chicken quarters.}


\begin{recipeabstract}
	A barbeque chicken collaboration.
	We prefer a 50 / 50 split between the two rubs for it's balanced flavor.
	The Pepper Rub works well on it's own for a traditional Barbeque, and the Chile Rub for tacos.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}[Single Deuce Pepper Rub]
		\ingredient[1][\Tablespoon]{peppercorns}\\
		\ingredient[1][\Tablespoon]{mustard, yellow}\\
		\ingredient[1][\Tablespoon]{onion powder}\\
		\ingredient[1][\Tablespoon]{garlic powder}\\
		\ingredient[1][\Tablespoon]{celery seed}\\
		\ingredient[1][\Tablespoon]{Montreal Steak mix}
	\end{ingredientblock}
	\vfill\null
	\columnbreak

	\begin{ingredientblock}[Single Deuce Chile Rub]
		\ingredient[1~\onehalf][\Tablespoon]{Guajillo powder}\\
		\ingredient[1][\Tablespoon]{salt, kosher}\\
		\ingredient[1][\Tablespoon]{paprika}\\
		\ingredient[\onehalf][\Tablespoon]{peppercorns}\\
		\ingredient[\onehalf][\Tablespoon]{nutmeg}\\
		\ingredient[\onehalf][\Tablespoon]{sage, ground}\\
		\ingredient[1][\teaspoon]{cayenne}\\
		\ingredient[1][\teaspoon]{brown sugar}
	\end{ingredientblock}
\end{ingredientcolumns}


\ragmarpar{You can eyeball each pepper rub ingredient instead of mixing it.}
\begin{preparation}
\item Prepare Pepper Rub.
	Coarsely grind pepper \& mustard, jar.
	Combine onion, garlic, celery, jar.
	Set aside Steak Seasoning.
\item Prepare chile rub.
	Grind the peppercorns, salt, sugar.
	Grind the nutmeg.
	Combine all, jar.
\item Rub chicken with pepper rub,
	using an equal mix of the garlic, peppercorns, and steak seasoning.
\item Rub chicken with chile rub,
	using approximately as much as the pepper rub.
\item Marinate for $\approx$ $[\onehalf~..~6]$ \hour.
\end{preparation}
