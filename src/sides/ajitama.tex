\begin{recipeabstract}
    Ramen Egg.
    A soft boiled egg cured in marinade, which flavors the egg and firms the yolk.
\end{recipeabstract}


\begin{ingredientcolumns}
    \begin{ingredientblock}
        \ingredient[6..8][]{egg}\\
        \ingredient[\onefourth][C]{mirin}\\
        \ingredient[\onefourth][C]{sake}
    \end{ingredientblock}
    \begin{ingredientblock}
        \ingredient[2][C]{water}\\
        \ingredient[\onehalf][C]{soy sauce}\\
        \ingredient[][]{dried shiitake}\\
    \end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\ragmarpar{Use an ice bath, not just cold water.}
\item Prepare \reffull{boiled_eggs} using a 6 $\minute$ boil.
\item Cool completely.
\item Bring mirin and sake to boil, flamb\'{e}.
\item Add soy sauce, water, optional dried aromatics.
\item Combine eggs \& marinade, refridgerate.
\item Marinate for 2 $\Day$.
\item Eat within 4 $\Day$.
\end{preparation}
