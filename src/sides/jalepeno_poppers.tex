\section{Jalepe\~{n}o Poppers}


\begin{recipestats}[
	servings=5,
	preptime=15 \minute,
	bakingtime=25 \minute,
	source=Jane,
]
\end{recipestats}


\begin{recipeabstract}
	A game day classic.
\end{recipeabstract}
\ragmarpar{Jane likes the dab of BBQ sauce.}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[8][]{jalepe\~{n}o}\\
		\ingredient[8][slice]{bacon}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[4][\ounce]{cream cheese}\\
		\ingredient[4][\ounce]{cheddar}\\
		\ingredient[][]{barbeque sauce}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Measure cream cheese and let soften.
\ragmarpar{Slicing bacon makes it easier to wrap.}

\item Line a baking sheet with foil, preheat oven to $425\Fahrenheit$.

\item Slice peppers in half, leaving stems intact.

\item Slice bacon in half lengthwise.

\item Shred cheddar, mix with cream cheese.

\item Fill the peppers with cheese mixture, wrap w/ bacon.
\ragmarpar{Wrap bacon similar to a clove hitch.}

\item Lightly brush with barbeque sauce.

\item Bake $\approx25$ $\minute$, or until bacon is done, brush w/ sauce.
\end{preparation}


\recipeend%
