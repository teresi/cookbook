\section[Acorn Squash]{Maple Bacon Acorn Squash}\index{squash!Acorn~Squash}


\begin{recipestats}[
    servings=12,
    preptime=15~\minute,
    bakingtime=40~\minute,
    source=Pamela Teresi,
	original=\citefield{squashCountryLiving2014}{journaltitle}~\cite{squashCountryLiving2014},
]
\end{recipestats}
\ragmarpar{Emma likes this.}


\begin{recipeabstract}
    A Teresi Thanskgiving tradition.
    It's easy to omit the bacon and use margarin for a vegetarian option.
\end{recipeabstract}


\begin{ingredientcolumns}
    \begin{ingredientblock}
        \ingredient[2][]{acorn squash}\\
        \ingredient[\onehalf][C]{maple syrup}\\
        \ingredient[\onefourth][C]{butter}\\
        \ingredient[2][\Tablespoon]{brown sugar}
    \end{ingredientblock}
    \begin{ingredientblock}
        \ingredient[2][\Tablespoon]{red vinegar}\\
        \ingredient[1][\Tablespoon]{rosemary}\\
        \ingredient[1][\teaspoon]{salt}
    \end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Line two baking sheets with parchment paper.
\item Preheat oven to 400 $\Fahrenheit$.
\item Slice squash into 1 $\inch$ rings, arrange into one layer.
\item Melt butter, add sugar and salt, brush onto squash.
\item Bake 20 $\minute$, flip, bake another 10 $\minute$.
\item Meanwhile, chop rosemary and mix maple syrup and vinegar.
\item Meanwhile, saut\'{e} bacon in a skillet until crisp.\\
    Remove bacon and drippings.
\item Add 1 $\Tablespoon$ bacon grease, add rosemary, cook for 1 $\minute$.
\item Add syrup and vinegar.\\
    Cook stirring often for 5 $\minute$, until reduced by half.
\item Move squash to serving platter, drizzle sauce.
\item Crumble bacon on top with extra rosemary and black pepper.
\end{preparation}


\recipeend%
