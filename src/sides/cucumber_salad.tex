\section{Cucumber Salad}\index{cucumber!Cucumber Salad}\index{kimchi!Cucumber Salad}


\ragmarpar{English cucumbers are best, peeling not required.}
\begin{recipestats}[
	servings=4,
	preptime=15~\minute,
	source=Mike \& Jane,
	original=Emma \& Josh,
]
\end{recipestats}


\begin{recipeabstract}
	Cucumbers jazzed up with soy sauce and vinegar.
	Works well in a rice bowl.
	The kimchi kicks it up a notch.
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[1][]{cucumber}\\
		\ingredient[1][\teaspoon]{salt}\\
		\ingredient[\approx\oneeighth][C]{kimchi}\\
		\ingredient[2][]{green onion}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[1][\Tablespoon]{soy sauce}\\
		\ingredient[1][\Tablespoon]{seasoned vinegar}\\
		\ingredient[1][\Tablespoon]{sesame seeds}\\
		\ingredient[1][\teaspoon]{sesame oil}
	\end{ingredientblock}
\end{ingredientcolumns}


\ragmarpar{Score cucumber sides w/ fork to help it retain the juice.}
\begin{preparation}
\item Slice cucumber, add to colander, toss w/ salt, let drain.
\item Meanwhile, slice onion, kimchi, mix all remaining.
\item Add cucumber, toss to mix.
\end{preparation}


\recipeend%
