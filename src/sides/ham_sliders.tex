\section[Ham Sliders]{Steve's Ham Sliders}\index{ham!Ham~Sliders}


\begin{recipestats}[
    servings=,
    preptime=30~$\min$,
    bakingtime=15~$\min$,
    source=Steven Teresi,
]
\end{recipestats}

\begin{recipeabstract}
    A Thanksgiving tradition.
    A crowd pleaser.
    Made as a side to eat while preparing the Turkey.
    ``The key is to drench it in butter.''
\end{recipeabstract}


\begin{ingredientcolumns}
    \begin{ingredientblock}
        \ingredient[1][loaf]{Hawaiin rolls}\\
        \ingredient[\onehalf][\pound]{swiss}\\
        \ingredient[\onefifth][\pound]{ham}\\
        \ingredient[1][medium]{white onion}\\
        \ingredient[][]{mayo}\\
        \ingredient[][]{Dijon}
    \end{ingredientblock}

    \bigskip
    \begin{ingredientblock}
        \ingredient[4][\Tablespoon]{butter}\\
        \ingredient[2][\teaspoon]{onion powder}\\
        \ingredient[1][\teaspoon]{Worcestershire}\\
        \ingredient[][]{minced onion}\\
        \ingredient[][]{poppy seed}
    \end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\ragmarpar{Always use the finger guard for the mandolin.}
\item Reserve a mandolin, grease a baking sheet.
\item Preheat oven to 325 $\Fahrenheit$.
\item Slice rolls in half way up the bottom.
\item Spread mayo thinly on one side, mustard on the other.
\item Slice onion on a mandolin, set aside.
\item Layer ham, swiss, half the onions.
\item Add the top half of the rolls.
\item Melt the butter with onion powder and Worcestershire.
\item Brush top with butter, top with spices and onions.
\item Bake for 10 $\minute$ or so until browned.
\end{preparation}
