\section[Baba Ganoush]{Baba Ganoush}\index{eggplant!Baba~Ganoush}\index{tahini!Baba~Ganoush}


\begin{recipestats}[
	servings=4,
	preptime=15~\minute,
	bakingtime=60~\minute,
	source=Mike \& Jane,
]
\end{recipestats}


\begin{recipeabstract}
	Excellent as a dip for baked pita chips and Falafel.
	The garlic confit really brings it all together.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1][large]{eggplant}\\
		\ingredient[\onehalf][C]{tahini}\\
		\ingredient[\onehalf][]{lemon, juice}\\
		\ingredient[4][clove]{garlic}\\
		\ingredient[2][\Tablespoon]{garlic confit}\\
		\ingredient[3][\Tablespoon]{olive oil}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[1][\teaspoon]{salt}\\
		\ingredient[1][\teaspoon]{cumin}\\
		\ingredient[1][\teaspoon]{fennel}\\
		\ingredient[1][\teaspoon]{onion powder}\\
		\ingredient[1][\teaspoon]{sumac}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Preheat to 425 $\Fahrenheit$, reserve a baking sheet, food processor.
\item Cut eggplant into quarters lengthwise, add to baking sheet, lightly salt, bake for $\approx$ $[30..45]$ $\minute$.
\item Meanwhile crush garlic, reserve sumac for garnish, grind the spices.
\item Let cool and scoop out eggplant from rind, add to processor.
\item Add tahini, garlic, garlic confit, spices, lemon juice.
\item Blend until mostly smooth, add olive oil while blending.
\item Spoon into a bowl, garnish with sumac and dash of olive oil.
\end{preparation}
