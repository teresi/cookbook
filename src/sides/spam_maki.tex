\section{Spam Maki}


\begin{recipestats}[
	servings=4,
	preptime=60~\minute,
	bakingtime=30~\minute,
	inactivetime=30~\minute,
	source=Mike \& Jane,
	original=\citetitle{alohaKitchen}~\cite[p.~42]{alohaKitchen},
]
\end{recipestats}


\begin{recipeabstract}
	A snazzy app.
	Jane and I had Spam musubi on our honeymoon and wanted to recreate that magic.
	We remixed ``Maki Sushi'' \cite{alohaKitchen} which calls for egg, carrot, tuna, watercress, \& gourd.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[4][C]{rice}\\
		\ingredient[\onethird][C]{rice vinegar}\\
		\ingredient[\onethird][C]{sugar}\\
		\ingredient[2][\teaspoon]{salt}\\
		\ingredient[1][\Tablespoon]{soy sauce}\\
		\ingredient[1][\Tablespoon]{mirin}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\onehalf][can]{Spam}\\
		\ingredient[1][]{cucumber}\\
		\ingredient[2][]{carrot}\\
		\ingredient[4][sheet]{nori}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
	\item Rinse and cook the sushi rice.
	\item Meanwhile, dissolve the vinegar, sugar, salt.
	\item When the rice is done, pour the vinegar over and fluff it.\\
		Let cool over a baking sheet.
	\ragmarpar{Use a mandolin for the vegetables.}
	\item Meanwhile, julienne cut the cucumber \& carrot.
	\item Slice Spam into disks, pan fry the Spam until lightly browned.
	\item Add soy \& mirin, cook to evaporate, cool, julienne.
	\item Wait until the rice is cool.
	\item Assemble. Lay the nori on the mat shiny side down. Spread a $\approx\onefourth~inch$ layer of rice and leave a $1 \onehalf~inch$ empty space of nori at the top.
	\item Lay the fillings in a thin layer starting $\approx1\onehalf~inch$ from the bottom.
	\item Roll the maki away from you, applying pressure as you go.
	\item Slice $\approx\onehalf~inch$ thick, fan out on a plate.
\end{preparation}


\recipeend%
