\section[Honey Ham Glaze]{Honey Mustard Ham Glaze}\index{honey!Honey~Ham~Glaze}


\begin{recipestats}[
	servings=6 \pound,
	preptime=10 \minute,
	bakingtime=10 \minute,
]
\end{recipestats}


\begin{recipeabstract}
	A classic honey mustard ham glaze.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[\onefourth][C]{butter}\\
		\ingredient[\onehalf][C]{brown sugar}\\
		\ingredient[\onefourth][C]{honey}\\
		\ingredient[\onefourth][C]{mustard, Dijon}\\
		\ingredient[1][\Tablespoon]{mustard, whole grain}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\oneeighth][\teaspoon]{cinnamon}\\
		\ingredient[\oneeighth][\teaspoon]{cloves}\\
		\ingredient[2][cloves]{garlic}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Measure and mix brown sugar, cinnamon, cloves.
	Measure honey.
	Smash and peel garlic.

\item Brown the butter in a small saucepan over medium low heat to a light copper color.

\item Add sugar, honey, spices, and mustard.
	Stir until dissolved.

\item Crush in garlic and stir.
	Cook for another few minutes and remove from heat.

\item Apply glaze once per $\approx 15$ \minute~in the last $\approx 45$ \minute~of cooking.
\end{preparation}


\recipeend%
