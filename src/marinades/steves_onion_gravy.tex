\section[Steve's Onion Gravy]{Steve's Onion Gravy}


\begin{recipestats}[
	servings=3~C,
	preptime=10~\minute,
	bakingtime=60~\minute,
	original=Steven Teresi \& Gary Burgess,
]
\end{recipestats}

\begin{recipeabstract}
	Steve's Thanksgiving Tradition.

\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[\onehalf][C]{butter}\\
		\ingredient[2][large]{onion, white}\\
		\ingredient[\onefourth][C]{flour}\\
		\ingredient[2][C]{stock}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[1][\teaspoon]{salt}\\
		\ingredient[1][\teaspoon]{pepper}\\
		\ingredient[1][\Tablespoon]{brandy}\\
		\ingredient[1][\Tablespoon]{white wine}\\
		\ingredient[1][\Tablespoon]{cream}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Heat skillet to medium w/ dash of oil.
\item Slice the onions, add to a skillet w/ salt \& pepper.
\item Saut\'{e} on low w/ $\approx~30$ $\minute$, stirring occasionally.
\item Meanwhile, measure the flour and stock, set aside.
\item Raise to medium low, melt the butter and let the water cook off for a few minutes.
\item Add the flour, stir constantly for $[15..25]$ $\minute$ to a light copper color.
\item Slowly add the stock while stirring until fully incorporated.
\item Set to low.
	Before serving, stir in brandy, wine, cream.
\end{preparation}
