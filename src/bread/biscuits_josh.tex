\section{Josh's Biscuits}\index{biscuit!Josh's Biscuits}


\begin{recipestats}[
	servings=12,
	preptime=30 \minute,
	bakingtime=15 \minute,
	source=Joshua Mollohan,
]
\end{recipestats}


\ragmarpar{Mixing manually will instill a greater sense of pride.}
\begin{recipeabstract}
	A fine biscuit with or without meat.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[2 \onehalf][C]{flour, all purpose}\\
		\ingredient[8][\Tablespoon]{butter, unsalted}\\
		\ingredient[1][\teaspoon]{salt}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[2][\Tablespoon]{baking powder}\\
		\ingredient[1][C]{buttermilk}\\
		\ingredient[\onefourth][C]{honey}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Preheat oven to $425~\Fahrenheit$ ($400~\Fahrenheit$ convention).

\ragmarpar{The dough will be very messy.}
\item Mix flour, baking powder, and salt in a food processor.

\item Cube the butter, pulse in food processor to a crumb texture.

\item Move mix to a bowl, create a hole in the center, and add buttermilk \& honey.
	Mix just enough to combine.

\item Turn dough onto a lightly floured surface.
	Roll or press dough to $\approx\onehalf''$, brush w/ buttermilk, then fold.
	Repeat $[3..5]$ times.

\item Cut biscuits out of dough w/ a glass or biscuit cutter.
	Do not twist cutter as this will ruin them and bring you shame.

\item Place on a baking sheet and brush w/ buttermilk.
\ragmarpar{Brushing w/ honey will appease small children.}

\item Bake $[15..18]$ \minute.

\item Optionally glaze w/ mix of $1~\Tablespoon$ honey \& $2~\Tablespoon$ melted butter.
\end{preparation}
