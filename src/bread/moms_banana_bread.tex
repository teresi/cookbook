\section[``Best'' Banana Bread]{Jean's ``Best'' Banana Bread}\index{banana!Best~Banana~Bread}\index{bread!Best~Banana~Bread}


\begin{recipestats}[
	servings=1 loaf,
	preptime=30 \minute,
	bakingtime=1 \onehalf~\hour,
	source=Pamela Teresi \& Jean Jesperse,
]
\end{recipestats}


\begin{recipeabstract}
	My friend Jean gave me this decades ago;
	she told Oma and I that it was the ``Best'' Banana Bread and we agreed.
\end{recipeabstract}
\ragmarpar{Buttermilk is best.}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1\onehalf][C]{banana, overripe}\\
		\ingredient[2][C]{flour}\\
		\ingredient[\threefourth][C]{butter}\\
		\ingredient[1\onehalf][C]{sugar}\\
		\ingredient[\onehalf][C]{buttermilk}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\onehalf][C]{nuts}\\
		\ingredient[2]{eggs}\\
		\ingredient[1][\teaspoon]{vanilla}\\
		\ingredient[1][\teaspoon]{salt}\\
		\ingredient[1][\teaspoon]{soda}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\ragmarpar{$\approx[4..6]$ bananas.}
\item Grease and line a loaf pan.
	Preheat oven to 350 \Fahrenheit.

\item Bring ingredients to room temperature.
	Sift flour, soda, salt.
	Mash the banana.
	Measure remaining dry ingredients.
\ragmarpar{Nuts are optional.}

\item Cream the butter.
	Add sugar gradually until combined.

\item Blend in banana, eggs, vanilla.

\ragmarpar{Add lemon juice to milk to mak sour milk.}
\item Add flour and milk alternately until just combined.
	Fold in nuts.

\item Pour into pan and bake at $350 \Fahrenheit$ for 1\onehalf~\hour.
\end{preparation}


\recipeend%
