\section[Lemonade Punch]{Sparkling Lemonade Punch}


\begin{recipestats}[
	servings=8,
	preptime=10~\minute,
	inactivetime=1~\hour,
	source=The~Gieskens,
]
\end{recipestats}

\begin{recipeabstract}
	A crowd pleaser.
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[12][\fluidounce]{pink lemonade concentrate, frozen}\\
		\ingredient[32][\fluidounce]{white cranberry juice}\\
		\ingredient[32][\fluidounce]{soda}\\
		\ingredient[8][\ounce]{strawberries}\\
		\ingredient[][]{mint}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Combine lemonade and juice, chill $[1 .. 24]$ $\hour$.
\item Add soda, sliced strawberries, mint; serve.
\end{preparation}


\recipeend%
