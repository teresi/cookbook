\section{Watermelon Agua Fresca}\index{watermelon!Watermelon~Agua~Fresca}


\begin{recipestats}[
	servings=16,
	preptime=15~\minute,
    original=\citeauthor{baylessWatermelon}\cite{baylessWatermelon},
]
\end{recipestats}


\begin{recipeabstract}
    Refreshing.
    Agua fresca is a sweetened cold drink, often including fruit, cereals, or flowers.
    Watermelon and lime are the main ingredients here.
    Serve over ice using a large beverage dispenser for pizzazz.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[8][C]{watermelon chunks}\\
        \ingredient[\onehalf][C]{lime juice}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[1][\quart]{water}\\
        \ingredient[1][C]{sugar}
	\end{ingredientblock}
\end{ingredientcolumns}


\ragmarpar{Adjust sweetness to taste, it should be less sweet than 100\% juice.}
\begin{preparation}
\item Reserve a blender and large dispenser, pitcher, or etc.
\item Cut watermelon into chunks.
\item Blend watermelon in batches with 1 $\onehalf$ $C$ water.
\item Press lime juice and blend in last batch with the sugar.
\item Stir in water and a few lime rinds.
\item Serve over ice.
\end{preparation}


\begin{variation}
\item Rick recommends $[7..8]$ $\pound$ cantaloupe, honeydew, papya, pineapple. Also recommends $[3.5..4]$ $\pound$ peaches, nectarines, guavas, and mangos.
    Add water as necessary.
\end{variation}
