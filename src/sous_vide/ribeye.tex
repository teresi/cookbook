\begin{recipeabstract}
	A classic steak recipe.
	Sous vide makes it easier to reach your target temperature and gives lee way for timing the rest of the meal.
\end{recipeabstract}
\ragmarpar{Also see~\cite{leoneBonAppetit2018} for technique.}

\begin{ingredientcolumns}
	\begin{ingredientblock}[sous vide]
		\ingredient[1][\pound]{ribeye, boneless 1''}\\
		\ingredient[1][\teaspoon]{salt}\\
		\ingredient[\onehalf][\teaspoon]{pepper, black}\\
		\ingredient[\onefourth][\teaspoon]{garlic powder}\\
		\ingredient[\onefourth][\teaspoon]{onion powder}
	\end{ingredientblock}
	\begin{ingredientblock}[searing]
		\ingredient[1][\Tablespoon]{oil}\\
		\ingredient[1][\Tablespoon]{butter}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Prepare water bath at $125$ \Fahrenheit.

\item Rub steak evenly with spices and dash of oil.
	Seal into bag.

\item Cook in water bath for $[40~\minute .. 2~\onehalf~\hour]$ (see variations).

\ragmarpar{No need to rest it.}
\item Remove steak onto plate w/ paper towel to dry.

\ragmarpar{Sear quickly, don't overcook.}
\item Open your window and turn on the range fan.
	Heat iron skillet on high w/ $\approx1~\Tablespoon$ oil until smoking.
	Melt the butter then lay steak into pan.
	Cook each side $\approx 30~\second$.
	Lift with tongs to cook each edge $\approx 15~\second$.

\item Remove and serve.
	Spoon some of the juice from the pan on top.

\end{preparation}


\begin{variation}
\item Cook 60 \minute~minimum for steaks $\approx1\onehalf~\inch$, 40 \minute~else.
\ragmarpar{Do not cook $>2\onehalf~\hour$ if $<130\Fahrenheit$ for food safety!}
	\begin{center}
	\begin{tabular}{r l l}
	rare        & $[120..129]$ \Fahrenheit & $[1..2\onehalf]~\hour$\\
	medium rare & $[130..134]$ \Fahrenheit & $[1..3\onehalf]~\hour$\\
	medium      & $[135..144]$ \Fahrenheit & $[1..3\onehalf]~\hour$\\
	medium well & $[145..155]$ \Fahrenheit & $[1..3\onehalf]~\hour$\\
	well        & $[156..165]$ \Fahrenheit & $[1..3\onehalf]~\hour$
	\end{tabular}
	\end{center}

\item Replace spices w/ Worcestershire sauce or pickle juice.

\item Add fresh thyme or rosemary.

\item Add butter at end of searing to reduce the char.
\end{variation}
