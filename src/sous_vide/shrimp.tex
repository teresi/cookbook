\section[Shrimp]{Shrimp}\index{shellfish!Shrimp}


\begin{recipestats}[
	servings=2,
	preptime=10 \minute,
	bakingtime=[15~\minute..1~\hour],
	source=Mike \& Jane,
	original=\citefield{anovaShrimp}{title}~\cite{anovaShrimp},
]
\end{recipestats}


\begin{recipeabstract}
	Perfect shrimp every time.
\end{recipeabstract}

\ragmarpar{Baking soda gives it a snappier texture.}
\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[1\onehalf][\pound]{shrimp}\\
		\ingredient[\onehalf][\teaspoon]{baking soda}\\
		\ingredient[\onehalf][\Tablespoon]{olive oil}\\
		\ingredient[1][\teaspoon]{cumin}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Prepare water bath at $135$ \Fahrenheit.

\item Add ingredients to a bag, shake, and seal.

\ragmarpar{Add shrimp to the tail end of a sous vide steak for surf \& turf.}
\item Cook $[15~\minute \ldots 1~\hour]$.
\end{preparation}


\begin{variation}
\item Change the temperature to tune the texture.
\begin{center}
\begin{tabular}{l l}
	translucent, soft \& buttery	& $[125]$ \Fahrenheit \\
	nearly opaque, very tender	& $[130]$ \Fahrenheit \\
	barely opaque, tender		& $[135]$ \Fahrenheit \\
	traditional poached, firm	& $[140]$ \Fahrenheit
\end{tabular}
\end{center}

\item Use oil or butter, and/or other spices.
\end{variation}


\recipeend%
