\section[Baked Ziti]{Lynn's Baked Ziti}\index{pasta!Lynn's~Baked~Ziti}


\begin{recipestats}[
	servings=10,
	preptime=15~\minute,
	bakingtime=45~\minute,
	original=\citeauthor{allRecipesZiti}~\cite{allRecipesZiti},

\begin{recipeabstract}
	A classic baked ziti.
	Made by Tara's family as a gift when half of the Gieskens fell ill.
	Lynn liked it so they shared the recipe, and now it is a Giesken classic.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1][\pound]{dry ziti}\\
		\ingredient[1][]{onion}\\
		\ingredient[1][\pound]{ground beef}\\
		\ingredient[52][\ounce]{spagetti sauce}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[6][\ounce]{provolone}\\
		\ingredient[1 \onehalf][C]{sour cream}\\
		\ingredient[6][\ounce]{mozzarella}\\
		\ingredient[2][\Tablespoon]{parmesan}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Dice the onion, shred the mozarella and parmasan cheese.
\item Grease a $9\times13$ $\inch$ baking dish, preheat oven to $350$ $\Fahrenheit$.
\item Brown the meat and onion over medium high,\\then stir in the sauce, simmer $15$ $\minute$.
\item Meanwhile bring a large pot of salted water to a boil.
\item Cook the pasta al dente and drain.
\item Layer into the dish $\onehalf$~ziti, provolone, $\onehalf$~sauce, $\onehalf$~ziti, mozzarella, $\onehalf$~sauce, parmasan.
\item Bake $30$ $\minute$, broil briefly for some color.
\end{preparation}
