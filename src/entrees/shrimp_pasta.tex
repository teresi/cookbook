\section[Shrimp Pasta]{Spicy Shrimp Pasta}


\ragmarpar{Jane likes this.}
\begin{recipestats}[
	servings=4,
	preptime=30~\minute,
	bakingtime=15~\minute,
	source=Jane \& Mike,
	original=\citeauthor{realSimpleShrimpPasta}~\cite{realSimpleShrimpPasta},
]
\end{recipestats}


\begin{recipeabstract}
	A rich buttery sauce on chewy pasta.
	Much too good for children.
	The Gochujang and sundried tomatoes really bring the sauce together.
	Gochujang has a unique flavor so don't skip it.
\end{recipeabstract}
\ragmarpar{Use sundried tomatoes in oil.}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[16][large]{shrimp}\\
		\ingredient[12][\ounce]{casarecce pasta}\\
		\ingredient[\onehalf][C]{butter}\\
		\ingredient[\onehalf][C]{sundried tomato}\\
		\ingredient[1][\Tablespoon]{Gochujang}\\
		\ingredient[2][\teaspoon]{paprika, smoked}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\onehalf][\teaspoon]{red pepper}\\
		\ingredient[\onehalf][\teaspoon]{salt, kosher}\\
		\ingredient[\threefourth][\teaspoon]{black pepper}\\
		\ingredient[1][\Tablespoon]{olive oil}\\
		\ingredient[][]{Asiago}
	\end{ingredientblock}
\end{ingredientcolumns}
\ragmarpar{Add a few cherry tomatoes.}


\begin{preparation}
\item Begin to boil salted water for the pasta.
\item Chop the tomatoes, peel and devein the shrimp.
\item Combine and set aside the butter, tomatoes, gochujang, paprika, red pepper, salt, black pepper.
\ragmarpar{Add tomato paste during saut\'{e}.}
\item Boil the pasta, drain and reserve \onehalf~$C$ of the water.
\item Heat some oil to medium high, add shrimp, cook $\approx$ 4 \minute.
\item Add butter mixture and half the reserved water.
	Stir well to emulsify until mostly evaporated.
\item Toss into pasta, add water to reach desired viscosity.
\ragmarpar{Mike adds chorizo.}
\item Garnish with black pepper and Asiago.
\end{preparation}
