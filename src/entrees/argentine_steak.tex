\begin{recipeabstract}
	Cumin and black pepper steak with a citrus couscous / spinach salad.
	Green scallions are used for a mellow Chimicchuri garnish.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[12][\ounce]{steak}\\
		\ingredient[1 \onefourth][\teaspoon]{cumin}\\
		\ingredient[\onehalf][\teaspoon]{pepper, black}\\
		\ingredient[\onefourth][\teaspoon]{salt}\\
		\ingredient[\threefourth][C]{couscous, Israeli}\\
		\ingredient[5][\ounce]{spinach}\\
		\ingredient[1][]{red bell pepper}\\
		\ingredient[3][cloves]{garlic}\\
		\ingredient[2][]{scallions, white}\\
		\ingredient[\onehalf][\teaspoon]{zest, lemon}
	\end{ingredientblock}
	\begin{ingredientblock}[Chimichurri]
		\ingredient[2][]{scallions, green}\\
		\ingredient[1][\Tablespoon]{oil, olive}\\
		\ingredient[\approx~1][\Tablespoon]{juice, lemon}
	\end{ingredientblock}
\end{ingredientcolumns}


\ragmarpar{Isreali couscous is pearled couscous.}
\begin{preparation}
\item Grind cumin, salt, pepper.
	Add a thin layer to steak with a dash of vegetable oil.
	Set aside, reserve remaining spice.

\item Begin boiling salted water for the couscous.
	Wash produce.

\item Slice bell pepper and scallion whites, mince scallion greens.
	Crush the garlic.
	Zest the lemon.
	Measure the spinach.

\ragmarpar{Also try parsley, oregano for chimichurri.}
\item Make the chimichurri.
	Mix scallion greens, oil, lemon juice in a small bowl.
	Season to taste w/ salt \& pepper.

\item Saut\`{e} bell pepper in a frying pan over medium heat for $\approx 5$ \minute.
	Add scallion whites, garlic; cook $\approx 1 \minute$.
	Remove, set aside.

\item Add couscous to boiling water, cook according to directions.
	When done, drain, and mix the couscous, spinach, bell pepper mix, lemon zest.
	Add olive oil / salt to taste.

\ragmarpar{Watch the steak carefully as the spices may burn.}
\item Meanwhile cook the steak.
	Sear $\approx [4 \dots 7]$ \minute~per side on medium heat, to $\approx 150$ \Fahrenheit.
	Set aside $\approx 5$ \minute~when done.

\item Thinly slice the steak.
	Plate on top of the couscous salad.
	Add a dash of the cumin mix and top w/ chimichurri.
\end{preparation}
