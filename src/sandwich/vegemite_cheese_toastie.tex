\section{Vegemite Cheese Toastie}


\begin{recipestats}[
	servings=1,
	preptime=5~\minute,
	bakingtime=15~\minute,
	source=Mike,
	original=\citefield{vegemiteCheeseToastie}{title}~\cite{vegemiteCheeseToastie},
]
\end{recipestats}


\ragmarpar{Mike likes this for the umami flavor.}
\begin{recipeabstract}
	A grilled cheese with Vegemite.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[2][slices]{bread}\\
		\ingredient[2][slices]{cheddar, sharp}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\approx~1][\teaspoon]{Vegemite}\\
		\ingredient[\approx~2][\Tablespoon]{butter}\\
		\ingredient[][]{mayo}
	\end{ingredientblock}
\end{ingredientcolumns}


\ragmarpar{Low heat is critical, do not burn!}
\begin{preparation}
\item Heat a frying pan on low, add butter and let melt.
\item One each slice, add a thin layer of Vegemite then the cheese.
\item Add bread, cheese face up.
\item Cover with lid, cook a few minutes until lightly browned, then close the sandwich.
\item Add a thin layer of mayo, flip, toast $\approx$ 1 $\minute$, repeat.
\end{preparation}


\recipeend%
