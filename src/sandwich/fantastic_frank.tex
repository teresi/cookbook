\begin{recipeabstract}
	Fanastic indeed.
	Jane's favorite sandwich, courtesy of Owens Hall.
	The freshly cut sides of the focaccia are flipped outwards so it gets extra toasty.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1]{focaccia bun}\\
		\ingredient[1][\Tablespoon]{pesto}\\
		\ingredient[1][\Tablespoon]{mayo}\\
		\ingredient[4][\ounce]{chicken breast}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[2][\ounce]{roasted red pepper}\\
		\ingredient[1 \onehalf][\ounce]{provolone}\\
		\ingredient{arugala}\\
		\ingredient{olive oil}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Cook and thinly slice chicken breast.

\item Reserve a cookie sheet and preheat to $\approx 400$ \Fahrenheit.
	Slice peppers.

\ragmarpar{$\approx 2$ slices provolone.}
\item Slice focacia in half, place cut ends down on the cookie sheet.
	Brush with olive oil.
	Top one side with provolone, then chicken.
	Toast $\approx 5$ \minute~or your preference.

\ragmarpar{Toast bread prior to baking for extra crunch.}
\item Meanwhile, prepare sauce with mayo \& pesto.

\item Top chicken with pepper, lettuce.
	Add sauce to bare focaccia.

\item Assemble sandwich, cut in half.
	Serve w/ pickle \& chips.
\end{preparation}
