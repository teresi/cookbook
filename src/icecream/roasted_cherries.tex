\section{Roasted Cherries}\index{cherry!Roasted~Cherries}

\begin{recipestats}[
        servings=1~\onefourth~C,
        bakingtime=30~\minute,
        inactivetime=1~\hour,
        original=\citetitle{jenisIcecream}~\cite[p.~196]{jenisIcecream},
    ]
\end{recipestats}

\begin{recipeabstract}
    Cherries in syrup for mixing and topping ice cream.
\end{recipeabstract}

\begin{ingredientcolumns}[1]
    \begin{ingredientblock}
        \ingredient[2][C]{cherries, fresh or frozen}\\
        \ingredient[\twothird][C]{sugar}\\
        \ingredient[2][\teaspoon]{cornstarch}
    \end{ingredientblock}
\end{ingredientcolumns}

\begin{preparation}
\item Preheat oven to 400 \Fahrenheit.
\item Mix all ingredients into a $9x9$ $\inch$~dish.
\item Bake $30 .. 40$ $\minute$, stirring every 15 $\minute$.
\item Cool, then refridgerate.
\end{preparation}

\recipeend%
