\section{Crisp Streusel}\index{oats!Crisp~Streusel}

\begin{recipestats}[
        servings=3~$C$,
        bakingtime=30~\minute,
        inactivetime=1~\hour,
        original=\citetitle{jenisIcecream}~\cite[p.~198]{jenisIcecream},
    ]
\end{recipestats}

\begin{recipeabstract}
    Crunchy oat crumb for mixing and topping ice cream.
\end{recipeabstract}

\begin{ingredientcolumns}
    \begin{ingredientblock}
        \ingredient[\onehalf][\pound]{butter, unsalted}\\
        \ingredient[1\onehalf][C]{flour}\\
        \ingredient[\oneeighth][\teaspoon]{cinnamon}
    \end{ingredientblock}
    \begin{ingredientblock}
        \ingredient[\threefourth][C]{brown sugar}\\
        \ingredient[\onehalf][\teaspoon]{salt}\\
        \ingredient[1\onehalf][C]{old-fashioned oats}
    \end{ingredientblock}
\end{ingredientcolumns}

\begin{preparation}
\item Reserve a large bowl, grease a cookie sheet.
    \ragmarpar{Crumbs should $\approx$ $\onefourth$ $\inch$}
\item Preheat oven to 350 \Fahrenheit.
\item Add all but the oats in a large bowl.
\item Rub butter into the dry ingredients to create crumbs.
\item Add the oats, mix. Move to cookie sheet.
    \ragmarpar{Freeze up to 1 $month$.}
\item Bake $30..35$ $\minute$, stirring occasionally.
\item Remove when lightly toasted.
\item Let cool. Freeze until ready to use.
\end{preparation}

\recipeend%
