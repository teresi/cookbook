\section[Cranberry Sorbet]{Cranberry Pomegranate Sorbet}


\begin{recipestats}[
	servings=12,
	preptime=15~\minute,
	bakingtime=45~\minute,
	inactivetime=8~\hour,
	original=\citefield{cranberrySorbetVegetarianBaker}{title}~\cite{cranberrySorbetVegetarianBaker},
]
\end{recipestats}


\begin{recipeabstract}
	A sour sorbet.
	We like this for Thanksgiving as it is a light after dinner desert, similar to the Thanksgiving lemon meringue.
	It's also vegan, so it's useful for family events.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[3][C]{cranberries}\\
		\ingredient[1][C]{Pomegranate seeds}\\
		\ingredient[2~\onehalf][C]{sugar}\\
		\ingredient[5][C]{water}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\onethird][C]{orange juice}\\
		\ingredient[\onehalf][C]{lemon juice}\\
		\ingredient[\onefourth][C]{lime juice}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Freeze your ice cream bowl 24 $\hour$ ahead.
\item Mix 2 $C$ water, $\onehalf$ $C$ sugar, cranberries, pomegranate, in a pot.
\item Heat over medium heat, bring to simmer, cook $\approx [10..15]$ $\minute$, or until the cranberries burst.
\item Move mixture to a blender, blend on high until smooth, strain well.
\item Chill in fridge $\approx [2..3]$ $\hour$.
\item Make simple syrup; add $2$ $C$ sugar and 3 $C$ water in a medium pot.
	Bring to simmer, dissolve completely, take off heat.
	Move to an airtight container and refridgerate.
\item Juice the lemons, limes, oranges.
	Mix juice, syrup, and cranberry mix in large bowl, whisk.
\item Freeze mixture in ice cream bowl according to those directions.
\item Move to freezer and let freeze $\ge2$ $\hour$.
\end{preparation}


\recipeend%
