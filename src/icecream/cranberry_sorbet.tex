\section[Pomegranate Sorbet]{Jane's Pom-Cran Sorbet}


\begin{recipestats}[
	servings=12,
	preptime=15~\minute,
	bakingtime=45~\minute,
	inactivetime=8~\hour,
	original=\citefield{cranberrySorbetVegetarianBaker}{title}~\cite{cranberrySorbetVegetarianBaker},
]
\end{recipestats}


\ragmarpar{You'll need $\onehalf$ a pomegranate, 12 $\ounce$ cranberries.}
\begin{recipeabstract}
	A tart sorbet.
	We like this for Thanksgiving as it's a light after dinner desert, similar to the Thanksgiving lemon meringue.
	It's useful for family events as it's vegan.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}[pomegranate juice]
		\ingredient[3][C]{cranberries}\\
		\ingredient[1][C]{Pomegranate}\\
		\ingredient[2][C]{water}\\
		\ingredient[\onehalf][C]{sugar}
	\end{ingredientblock}

	\bigskip
	\begin{ingredientblock}[syrup]
		\ingredient[2][C]{sugar}\\
		\ingredient[3][C]{water}
	\end{ingredientblock}
	
	\columnbreak
	\begin{ingredientblock}[citrus juice]
		\ingredient[\onethird][C]{orange juice}\\
		\ingredient[\onehalf][C]{lemon juice}\\
		\ingredient[\onefourth][C]{lime juice}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Freeze your ice cream bowl $\left[24 .. 48\right)$ $\hour$ ahead.
\item Reserve two small pots and a blender.
\item Make simple syrup; add $2$ $C$ sugar and 3 $C$ water to a pot.\\
	Heat on medium, dissolve completely, take off heat.
\item Meanwhile, seed the pomegranate, wash the cranberries.
\item Add 2 $C$ water, $\onehalf$ $C$ sugar, pomegranate, cranberries to a pot.
\item Simmer over medium for $\approx [10..15]$ $\minute$, or until cranberries burst.
\item Move to a blender, blend on high until smooth, strain well.
\item Let syrup \& pomegranates cool; refridgerate $\approx \left[2..3\right)$ $\hour$ or overnight.
\item Juice the lemons, limes, oranges.\\
	Mix juice, syrup, and pomegranate mix in large bowl, whisk.
\item Freeze mixture in ice cream bowl according to those directions.
\item Move to freezer and let freeze $\ge2$ $\hour$.
\end{preparation}
