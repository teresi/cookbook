\section[Buckeye Ice Cream]{The Buckeye State Ice Cream}\index{peanut~butter!Buckeye~Ice~Cream}

\begin{recipestats}[
        servings=16,
        preptime=15~\minute,
        bakingtime=45~\minute,
        inactivetime=4~\hour,
        original=\citetitle{jenisIcecream}~\cite[p.~104]{jenisIcecream},
    ]
\end{recipestats}

\begin{recipeabstract}
    ``The'' Buckeye Ice Cream.
    A chocolate and peanut butter mix.
    The chocolate is added while liquid to create flakes.
\end{recipeabstract}

\ragmarpar{Use unsalted peanut butter.}
\begin{ingredientcolumns}
    \begin{ingredientblock}
        \ingredient[2~\twothird][C]{whole milk}\\
        \ingredient[5][\teaspoon]{corn starch}\\
        \ingredient[2][\ounce]{cream cheese}\\
        \ingredient[\twothird][C]{peanut butter}\\
        \ingredient[\onehalf][\teaspoon]{salt}
    \end{ingredientblock}
    \begin{ingredientblock}
        \ingredient[1~\onehalf][C]{heavy cream}\\
        \ingredient[\threefourth][C]{sugar}\\
        \ingredient[2][\Tablespoon]{corn syrup}\\
        \ingredient[\onefourth][C]{honey}\\
        \ingredient[5][\ounce]{dark chocolate}
    \end{ingredientblock}
\end{ingredientcolumns}

\begin{preparation}
\item Freeze your ice cream bowl 1 or 2 $\Day$ ahead.
\item Reserve a sauce pot, 3 bowls, and a 1 $\gallon$~plastic bag.
\item Mix the cornstarch with $\approx 2$ $\Tablespoon$ of milk in a small bowl.
\item Whisk in a medium bowl:\\cream cheese, peanut butter, salt.
\item Fill a large bowl with ice and water.
\item Add to a sauce pot:\\cream, milk, sugar, corn syrup, honey.
\item Bring to a rolling boil over medium-high, boil 4 $\minute$.
\item Remove from heat, gadually mix in cornstarch.
\item Bring back to boil over medium high,\\cook while stirring $\approx~1$ $\minute$, until thickened.
\item Remove from heat.
\item Gradually whisk hot milk into cream cheese until smooth.
\item Pour into a 1 $\gallon$~plastic bag, submerge in ice bath.
\item Wait until it's cool, $\approx 30$ $\minute$.
    \ragmarpar{It's done when the ice cream pulls away from the sides of the freezer bowl.}
\item Transfer milk to freezer bowl and start mixing.
\item Meanwhile, melt chocolate in double boiler.
\item Remove from heat, let cool but keep don't let it set.
\item Spin until it's almost finished. \\
    Slowly drizzle chocolate on top.
\item Spin about $\approx 2$ $\minute$ until it's well mixed.
\item Pack into a container.
\item Press a sheet of parchment paper on top of ice cream.
\item Freeze in freezer $\ge4$ $\hour$.
\end{preparation}

\begin{experiments}
\item Try replacing the honey, it seems out of place.
\end{experiments}
