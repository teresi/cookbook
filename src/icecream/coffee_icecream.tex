\section[Coffee Icecream]{Coffee Nutella Swirl Icecream}\index{coffee!Coffee~Icecream}\index{Nutalla!Coffee~Icecream}


\begin{recipestats}[
	servings=1~quart,
	preptime=10 \minute,
	bakingtime=60 \minute,
	inactivetime=24 \hour,
	source=Mike \& Jane,
	original=\citefield{joyofcooking2006}{title}~\cite{joyofcooking2006},
]
\end{recipestats}


\ragmarpar{Kevin likes this for the fusion of flavors.}
\begin{recipeabstract}
	An elusive elixer with a saccherine swirl.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1~\onehalf][C]{whole milk}\\
		\ingredient[\threefourth][C]{sugar}\\
		\ingredient[\oneeighth][\teaspoon]{salt}\\
		\ingredient[3][large]{egg yolks}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[2][C]{heavy cream}\\
		\ingredient[1][\teaspoon]{vanilla extract}\\
		\ingredient[\onefourth][C]{coffee beans}\\
		\ingredient[\onefourth][C]{Nutella}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Add milk, sugar, salt, to a medium saucepan.
	Heat on medium low to a simmer while stirring to dissolve.
\item Coarsley grind the beans.
	Remove milk from heat, add the beans cover and steep 30 \minute.
\item Strain into a new saucepan, bring back to a simmer.
\ragmarpar{Decant custard to remove coffee dust.}
\item Separate egg yolks into a bowl and whisk.
\item Slowly whisk in $\approx\onehalf$ $C$ hot milk into yolks.
\item Slowly whisk eggs back into holt milk.
\item Reduce to low, stir constantly until it reaches $175$ $\Fahrenheit$.
	Do \textit{not} boil.
\ragmarpar{Slowly drizzle Nutella to achieve a swirl.}
\item Refridgerate until cold.
\item Mix the cream, vanilla, and custard.
	Prepare the ice cream as directed.
\item Scoop ice cream into a container, alternating w/ Nutella, freeze.
\end{preparation}
