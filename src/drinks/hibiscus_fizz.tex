\section[Hibiscus Fizz]{Hibiscus Fizz~\ragStars{3}}\label{hibiscus_fizz}\index{hibiscus!Hibiscus~Fizz}


\begin{recipestats}[
	servings=1,
	preptime=5~\minute,
	original=Mike \& Jane,
]
\end{recipestats}


\begin{recipeabstract}
    Fizzy and revitalizing.
	A simple and low alcohol use for \reffull{hibiscus_liqueur}.
\end{recipeabstract}


\ragmarpar{Use Topo Chico for pizzazz.}
\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[\threefourth][\fluidounce]{hibiscus liqueur}\\
		\ingredient[\onehalf][\fluidounce]{lime juice}\\
		\ingredient[\onehalf][\fluidounce]{simple syrup}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[4][\fluidounce]{seltzer}\\
		\ingredient[1][]{cherry}
	\end{ingredientblock}
\end{ingredientcolumns}


\ragmarpar{Optionally shake all but the seltzer.}
\begin{preparation}
\item Build in a highball crushed ice, hibiscus, lime, and syrup.
\item Top with seltzer, stir, fill with ice, garnish with a cherry.
\end{preparation}


\recipeend%
