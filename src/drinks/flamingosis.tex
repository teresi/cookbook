\section[Flamingosis]{Flamingosis, or,\\Pink Lemonade~\ragStars{3}}%\index{hibiscus!Flamingosis}\index{lemon!Flamingosis}


\begin{recipestats}[
    servings=1,
    preptime=5~\minute,
    original=Mike \& Jane,
]
\end{recipestats}


\begin{recipeabstract}
    Refreshing and tart.
    simple and low alcohol use for \reffull{hibiscus_liqueur}.
    The fragrant hibiscus pairs well with the sour lemon.
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[\onehalf][\fluidounce]{hibiscus liqueur}\\
		\ingredient[10][\fluidounce]{lemonade}
	\end{ingredientblock}
\end{ingredientcolumns}

\ragmarpar{Add simple syrup to taste if your lemonade is sour.}
\begin{preparation}
\item Fill a collins glass with crushed ice.
\item Add the syrup, then lemonade, stir. Top with a cherry.
\end{preparation}


\recipeend%
