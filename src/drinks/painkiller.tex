\begin{recipestats}[
	servings=1,

\end{recipestats}


\begin{recipeabstract}
	A trademarked drink originally from the Soggy Dollar Bar, then remixed by the founder of Pusser's Rum.
\end{recipeabstract}


\ragmarpar{Add a little float of spiced rum before the nutmeg.}
\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[2][\fluidounce]{Pusser's Rum}\\
		\ingredient[4][\fluidounce]{pineapple juice}\\
		\ingredient[1][\fluidounce]{orange juice}\\
		\ingredient[1][\fluidounce]{cream of coconut}\\
		\ingredient[][]{nutmeg}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Fill a goblet or double old fashioned with crushed ice.
\item Shake all ingredients except nutmeg with ice.
\item Strain, garnish w/ grated nutmeg, cherry, orange slice.
\end{preparation}


\recipeend%
