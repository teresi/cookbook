\section{SC Hibiscus Liqueur}\label{hibiscus_liqueur}\index{hibiscus!Hibiscus~Liquer}


\begin{recipestats}[
	servings=16~\fluidounce,
	preptime=5~\minute,
	original=\citeauthor{smugglersCove}~\cite[p.~331]{smugglersCove},
]
\end{recipestats}


\begin{recipeabstract}
	A deep magenta liqueur of hibiscus.
	Sweet, spicy, saturated.
	Used to make \reffull{hibiscus_fizz}.  % TODO add hibiscus rum punch
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[1~\onefourth][C]{\nameref{hibiscus_infusion}}\\
		\ingredient[1][C]{2:1 simple syrup}\\
		\ingredient[\onehalf][C]{honey}
	\end{ingredientblock}
\end{ingredientcolumns}
\ragmarpar{$\approx$ 170 $g$ honey.}


\begin{preparation}
	\item Mix all ingredients, let dissolve.
	\item Bottle and refrigerate. Keep up to 1 $month$.
\end{preparation}


\recipeend%
