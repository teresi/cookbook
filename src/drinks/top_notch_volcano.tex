\section[Top Notch Volcano]{Top Notch Volcano~\ragStars{3}}


\begin{recipestats}[
	servings=4,
	preptime=10~\minute,
	original=\citeauthor{smugglersCove}~\cite[p.~134]{smugglersCove},
]
\end{recipestats}


\begin{recipeabstract}
	A crowd pleaser with plenty of lime.
	The passion fruit is pleasantly sour brings it all together.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[4][\fluidounce]{lime juice}\\
		\ingredient[4][\fluidounce]{pineapple juice}\\
		\ingredient[1][\fluidounce]{passion fruit puree}\\
		\ingredient[3][\fluidounce]{demerara syrup}
	\end{ingredientblock}

	\begin{ingredientblock}
		\ingredient[1][\fluidounce]{Maraschino liqueur}\\
		\ingredient[4][\fluidounce]{blended lightly aged rum}\\
		\ingredient[4][\fluidounce]{blended aged rum}
	\end{ingredientblock}
\end{ingredientcolumns}


\ragmarpar{Try using an immersion blender in a pitcher instead of the flash blender and scorpion bowl.}
\begin{preparation}
	\item Prepare a $\approx[32..48]~\fluidounce$ scorpion bowl \& two mixing tins.
	\item Add all to a mixer tin w/o ice,\\
		pour back and fourth with another to mix.
	\item Divide evenly among the two tins, add to each $12\fluidounce$ of crushed ice and $[4..6]$ agitator cubes.
	\item Flash blend and open pour w/ gated finish into a scorpion bowl.
	\item Serve with 20 \inch~luau straws and garnish with citrus wheels.
	\item Ignite the volcano with \reffull{TikiFire}.
\end{preparation}

\recipeend%
