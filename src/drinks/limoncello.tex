\section[Limoncello]{Limoncello~\ragStars{3}}\index{lemon!Limoncello}

\begin{recipestats}[
        servings=1\onehalf~\gallon,
        preptime=1~\hour,
        inactivetime=4~\Month,
        source=Emma \& Josh,
    ]
\end{recipestats}

\begin{recipeabstract}
    An Italian treat.
    Serve cold after dinner and relax.
    Josh and Emma like to make this for the annual summer celebration.
    Try mixing with Prosecco, Gin, or \reffull{hibiscus_liqueur}.
\end{recipeabstract}

\begin{ingredientcolumns}
    \begin{ingredientblock}[tincture]
        \ingredient[15][]{lemons}\\
        \ingredient[1][\gallon]{vodka, $80^\circ$}
    \end{ingredientblock}

    \medskip
    \begin{ingredientblock}[syrup]
        \ingredient[6][C]{sugar}\\
        \ingredient[6][C]{water}\\
        \ingredient[\onehalf][\teaspoon]{vanilla extract}
    \end{ingredientblock}
\end{ingredientcolumns}

\begin{preparation}

\item Reserve a gallon carboy and six $32~\fluidounce$ swing-top bottles.

\item Reserve about 4 months of preparation time.

    \ragmarpar{Minimize the pith when zesting.}
\item Wash, zest the lemons, add zest to carboy with spirit.

\item Store in a cool dark place for $[2..3]~\Month$.

\item Prepare the $1:1$ simple syrup.

\item Strain the vodka into the bottles with the syrup,\\
    about $1\onehalf~C$ syrup and $2\onehalf~C$ vodka.

\item Store in a cool dark place for $[3..4]~\week$.

\item Freeze at least 2 days ahead of serving.

\item Store in the freezer.

\end{preparation}

\recipeend%
