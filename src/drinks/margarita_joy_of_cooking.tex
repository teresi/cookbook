\begin{recipestats}[
	servings=2,
	preptime=5 \minute,
	source=Mike \& Jane,
	original=\citefield{joyofcooking2006}{title}~\cite{joyofcooking2006},
]
\end{recipestats}
\ragmarpar{Jane likes this.}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[3][\fluidounce]{tequila, white}\\
		\ingredient[1][\fluidounce]{Cointreau}\\
		\ingredient[1][\fluidounce]{lime juice}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\ragmarpar{Prefer Jose Cuervo Tradicional over Especial.}
\item Shake ingredients with $\approx1$ $C$ ice for $\approx [10 .. 15]$ $\second$.
\item Rub outside edge of highball w/ lime, dip into salt.
\item Strain over a few ice cubes.
\end{preparation}


\recipeend%
