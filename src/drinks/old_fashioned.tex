\begin{ingredientcolumns}[1]

		\ingredient[2][\fluidounce]{bourbon, or rye}\\
		\ingredient[2][\teaspoon]{sugar}\\
		\ingredient[3][dash]{Angostura}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\onefourth][\fluidounce]{water}\\
		\ingredient[1][large]{orange peel}
	\end{ingredientblock}

\ragmarpar{Jane adds a cherry.}
\begin{preparation}
\item Add sugar, water, bitters, to a highball, muddle until mostly dissolved.
\item Add a few ice cubes, whiskey, express the orange peel, add.
\end{preparation}
