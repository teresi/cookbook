\section[Original Margarita]{Original Margarita\ragStars{3}}\index{lime!Original~Margarita}\index{tequila!Original~Margarita}


\begin{recipestats}[
	servings=1,
	preptime=5 \minute,
	source=Mike \& Jane,
	original=\citefield{cointreauMargarita}{shorttitle}~\cite{cointreauMargarita},
]
\end{recipestats}
\ragmarpar{Mike likes this.}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[2][\fluidounce]{tequila, white}\\
		\ingredient[1][\fluidounce]{Cointreau}\\
		\ingredient[1][\fluidounce]{lime juice}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[1][small]{lime peel}\\
		\ingredient[1][dash]{Agave syrup}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\ragmarpar{Prefer Jose Cuervo Tradicional over Especial}
\item Shake ingredients with $\approx1$ $C$ ice for $\approx [10 .. 15]$ $\second$.
\item Open pour into a highball.
\end{preparation}


\begin{variation}
\item Substitue Cointreau with Triple Sec, dry cura\c{c}ao not recommended.
\end{variation}


\recipeend%
