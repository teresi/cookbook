\ragmarpar{Fresh nutmeg is critical.}

\begin{recipeabstract}
	A Smuggler's Cove original.
	Similar to a Daiquiri in that it's primarily rum, sugar, lime; but more complex due to the aged rum, dark sugar, and the addtion of bitters and nutmeg.
\end{recipeabstract}

% TODO update SC Demerara syrup to a link

\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[1][\fluidounce]{lime juice}\\
		\ingredient[1][\fluidounce]{SC Demerara syrup}\\
		\ingredient[2][\fluidounce]{blended aged rum}\\
		\ingredient[1][dash]{Angostura}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
	\item Fill a footed pilsner with cubed or cracked ice.
		\ragmarpar{Add straw, then nutmeg, \& extra dash of bitters.}
	\item Add ingredients to shaker, fill with cubed or cracked ice.
	\item Shake \& strain into glass, top w/ grated nutmeg.
\end{preparation}
