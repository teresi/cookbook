\ragmarpar{Mike likes this.}
\begin{recipeabstract}
	A variation of the \it{Jungle Bird} by Giuseppe Gonz\'{a}lez, deemed the very best by \it{PUNCH} magazine\cite{originalJungleBird}.
	Giuseppe notably prefers flavored black rum over the original dark Jamaican rum, and reduced the pineapple juice.
\end{recipeabstract}

\ragmarpar{Giuseppe prefers Cruzan Black Strap\cite{originalJungleBird}.}
\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[1\onehalf][\fluidounce]{pineapple juice}\\
		\ingredient[1\onehalf][\fluidounce]{black rum}\\
		\ingredient[\threefourth][\fluidounce]{Campari}\\
		\ingredient[\onehalf][\fluidounce]{lime juice}\\
		\ingredient[\onehalf][\fluidounce]{simple syrup}
	\end{ingredientblock}
\end{ingredientcolumns}

\begin{preparation}
\item Shake with ice.
\item Strain into a rocks glass over one giant ice cube.
\item Garnish with an orange slice.
\end{preparation}


\begin{variation}
\item Try \emph{dark} or \emph{black} rum.
	The name of \emph{dark} / \emph{black} / \emph{blackstrap} rum is ambiguous.
\ragmarpar{All black rums are dark, but not all dark rums are black.}
	Confusingly, there is no such thing as `Blackstrap Rum' as all rum is made with some blackstrap mollasses;
	the fundamental difference is whether the color and flavor came from barrel aging or backsweetening\cite{whatIsBlackstrapRum}.
	This backsweetening uses coloring and flavorings to imitate the look and taste of molasses.
	\emph{Dark} rum typically refers to aged, while \emph{black} is sweetened.
	Thankfully the author also specifies ``Cruzan Black Strap'' is preferred\cite{originalJungleBird}\cite{giuseppeJungleBird}, meaning the sweetened rum should be used.
\item Mike prefers to cut the Campari to \onehalf~\fluidounce~, replace the simple syrup with agave, and add a lime peel to the shaker.
\end{variation}
