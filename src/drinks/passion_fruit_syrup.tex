\section{Passion Fruit Syrup}\label{passionFruitSyrup}\index{passion fruit!Passion Fruit Syrup}


\begin{recipestats}[
	servings=3~$C$,
	preptime=10~\minute,
	original=\citeauthor{smugglersCove}~\cite[p.~325]{smugglersCove},
]
\end{recipestats}

\begin{recipeabstract}
	Rich syrup with passion fruit.
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[1\onehalf][C]{passion fruit pulp}\\
		\ingredient[1\onehalf][C]{2:1 simple syrup}
	\end{ingredientblock}
\end{ingredientcolumns}
\ragmarpar{See \reffull{richSyrup}}


\begin{preparation}
	\item Let rich syrup cool, whisk fruit puree and syrup.
	\item Bottle and store in refridgerator $\le 10 \Day$.
\end{preparation}


\recipeend%
