\section{Tiki Fire}\label{TikiFire}\index{fire!Tiki~Fire}


\begin{recipestats}[
	servings=1,
	preptime=5~\minute,
	original=\citeauthor{smugglersCove}~\cite[p.~245]{smugglersCove},
]
\end{recipestats}

\begin{recipeabstract}
	A ``crowd pleaser'' garnish with a dramatic bright yellow flame.
	This is preferred over 151 rum dispensed into a spent lime shell, which produces only a ``sad blue flame''.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1]{crouton, plain}\\
		\ingredient{lemon extract}
	\end{ingredientblock}

	\begin{ingredientblock}
		\ingredient{cinnamon powder}\\
		\ingredient{nutmeg powder}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
	\item Soak crouton in lemon extract then float upon a citrus wheel.
	\item Direct your guests' attention to the impending flame.
	\item Light crouton, shake the spices above it to produce sparks.
	\item Extinguish or remove the flame before imbibing.
\end{preparation}


\recipeend%
