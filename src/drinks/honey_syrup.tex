\section{Honey Syrup}\index{honey!Honey~Syrup}


\begin{recipestats}[
	servings=$1~C$,
	preptime=5~\minute,
	source=Mike \& Jane,
	original=\citefield{cocktailSeminars2021}{title} \cite{cocktailSeminars2021},
]
\end{recipestats}


\begin{recipeabstract}
	Honey diluted with water to improve solubility.
	This form is easier to measure and mix with cold liquids.
	Approximately $\threefourth~C$ honey to $\onefourth~C$ water.
\end{recipeabstract}


\ragmarpar{$\threefourth~\fluidounce$ of this can replace $\onehalf~\fluidounce$ honey\cite{cocktailSeminars2021}.}
\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[\approx250][\gram]{honey}\\
		\ingredient[60][g]{water}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Microwave water to lukewarm, transfer to bottle.
\item Add honey and shake until dissolved.
\item Refrigerate up to $3~weeks$.
\end{preparation}


\recipeend
