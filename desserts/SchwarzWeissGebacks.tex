\begin{recipe}
        [
preparationtime = 30 minutes prep \& 1.5 hour chill,
bakingtime = 12-15 minutes per batch,
bakingtemperature = {\unit[350]{$F^\circ$}},
portion = $\sim$ 40 cookies,
%calory = $\sim$ 125 each,
source = Classic German Baking (Lusia Weiss 2016)
]{Schwarz-Weiss G\"{e}back}
        \graph{
%small = strawberry,
smallpicturewidth = 0.3\textwidth,
%big = strawberrycake,
bigpicturewidth = 0.6\textwidth,
}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\ingredients{
\unit[10.5]{Tb} & unsalted butter (150g)\\
\unit[9]{Tb} & confectioners sugar (75g)\\
\unit[1/8]{ts} & salt\\
\unit[1/4]{ts} & vanilla extract\\
\unit[1 2/3]{Cup} & all purpose flour (200g)\\
\unit[1]{} & egg yolk\\
\unit[2]{Tb} & whole milk\\
\unit[2 1/2]{Tb} & cocoa powder
}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\preparation{
\step Cream butter $\sim 1 minute$, add sugar, salt, vanilla, then cream.
Add flour and mix until just combined.
\step Take 1/2 dough, make a disk, wrap in plastic wrap.
Mix cocoa into other half of dough, wrap as well, refrigerate both $\sim 1 hour $.
\step Combine milk \& egg yolk in a small bowl.
\step Make 4 square logs, brush sides w/ egg wash, press together and refrigerate $\sim 30 minutes$.
\step Preheat oven \& line baking sheets with parchment paper while chilling.
\step Slice off cookies to $1cm$ or other desired thickness, bake $12-15 minutes$
}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\hint{
\begin{itemize}
\item Try other designs such as pinwheels, 3x3 checkerboards, ying-yangs, marble, stripes, etc.
\item Try dipping bottom in chocolate.
\item Try arranging little balls into 6 sided flowers instead of slicing.
\item Try not to over bake for aesthetics.
\end{itemize}
}
