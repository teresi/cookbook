\begin{recipe}[
source = Lucille Steinmiller (Oma)
]
{Elegant Wine Cake}

\ingredients {
\unit[1]{pkg} & cake mix, yellow\\
\unit[1]{pkg} & vanilla pudding, instant\\
\unit[1/2]{C} & vegetable oil\\
\unit[4]{} & eggs\\
\unit[3/4]{C} & sherry, medium\\
\unit[1/4]{C} & water\\
\unit[1]{C} & chopped nuts
}

\preparation {
\step Combine all ingredients in a bowl, beat for 2 minutes on medium speed.
\step Bake in 2 loaf pans $\approx 9x5"$ (greased and floured) at $350\; F^\circ$ for $45\; minutes$ or until done.
\step Top with powdered sugar.
}

\hint{
\begin{itemize}
\item We usually use pecans or omit it. Chablis can be subbed for the sherry.
\item Individual cake or a bundt pan also works well.
\end{itemize}
}

\end{recipe}
