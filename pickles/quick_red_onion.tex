\section[Pickled Red Onions]{Quick Pickled Red Onions}\label{pickled_red_onions}

\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[4][cloves]{garlic}\\
		\ingredient[\approx 2][medium]{red onions}\\
		\ingredient[\onehalf][\teaspoon]{oregano, mexican}\\
		\ingredient[\onehalf][\teaspoon]{cumin seeds}\\
		\ingredient[\onehalf][\teaspoon]{mustard seeds, brown}\\
		\ingredient[1][\teaspoon]{pequin chili, dry}\\
		\ingredient[\onehalf][\teaspoon]{allpsice}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Cut onions in half across the axis, then into $\approx\onefourth "$ slices across the axis.
	Smash, then coarsley slice the garlic.

\item Use method \ref{basic_quick_pickles} \nameref{basic_quick_pickles}, replacing the vegetables and spices with this recipe.
\end{preparation}
