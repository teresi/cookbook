\section[Red No. 4]{Red No. 4~\vegan}


\begin{recipestats}[
	servings=12 \fluidounce,
	preptime=1 \hour,
	bakingtime=4 \week,
	source=Teresi Family,
	]
\end{recipestats}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[220][g]{habenero, red}\\
		\ingredient[40][g]{cabbage}\\
		\ingredient[30][g]{carrot}\\
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\onefourth][\teaspoon]{xanthan gum}\\
		\ingredient[\onethird][\fluidounce]{lime juice, bottled}\\
		\ingredient[\onefourth][\teaspoon]{achiote}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Follow~\ref{base_lacto_brine_hotsauce}~\nameref{base_lacto_brine_hotsauce} with a $5.3\%$ brine and the habenero, cabbage, carrot.

\item Ferment 2 \week, refridgerate 2 \week.

\item Blend vegetables with $\onehalf~C$ brine, $\onehalf~C$ vinegar, for $5~\minute$.
	Strain through a mesh seive, rinse blender, move sauce back to blender.

\item Add lime juice, achiote.
	Turn on blender and slowly add the xanthan gum.
\end{preparation}


\begin{experiments}
\item \onefourth~\teaspoon~xanthan gum produced an ok consistency but try \oneeighth~next.

\item Increase lime juice to 1 \ounce?
\end{experiments}


\recipeend%
