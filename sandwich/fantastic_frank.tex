\section{Fantastic Frank}


\begin{recipestats}[
	servings=1,
	preptime=10 \minute,
	bakingtime=10 \minute,
	source=Virginia Tech,
]
\end{recipestats}


\begin{recipeabstract}
	Fanastic indeed.
	Jane's favorite sandwich, courtesy of Owens Hall.
	The freshly cut sides of the foccocia are flipped outwards so it gets extra toasty.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1]{foccocia bun}\\
		\ingredient[\onehalf][\teaspoon]{pesto}\\
		\ingredient[1][\Tablespoon]{mayo}\\
		\ingredient[4][\ounce]{chicken breast}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[2][\ounce]{roasted red pepper}\\
		\ingredient[1 \onehalf][\ounce]{provolone}\\
		\ingredient{mixed lettuce}\\
		\ingredient{olive oil}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Cook and cut chicken breast to $\approx \onehalf "$ slices.

\item Reserve a cookie sheet and preheat to $\approx 400$ \Fahrenheit.
	Prepare sauce with mayo \& pesto.
	Slice red peppers.

\ragmarpar{$\approx 2$ slices provolone}
\item Slice foccocia in half, place cut ends down on the cookie sheet.
	Brush with olive oil.
	Top one side with provolone, then chicken.
	Toast $\approx 5$ \minute~or your preference.
	Remove.

\item Top chicken with pepper, lettuce.
	Add sauce to bare foccocia.
	Assemble sandwich with toothpicks, cut in half.
	Serve with dill pickle, potato chips.

\end{preparation}


\begin{variation}
\item Toast bread prior to baking for extra crunch.
\end{variation}


\recipeend
