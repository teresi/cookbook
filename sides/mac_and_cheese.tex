\section[Classic Mac]{Classic Macaroni and Cheese}
\index{pasta!Classic Mac}
\index{cheese!Classic Mac}


\begin{recipestats}[
	servings=8 people,
	preptime=25 \minute,
	bakingtime=25 \minute,
	source=The Gieskens,
	]
\end{recipestats}


\begin{recipeabstract}
	This is a classic baked macaroni and cheese that is fast and perfect for parties.
	The secret ingredient is Dijon mustard.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[3][C]{macaroni}\\
		\ingredient[\nicefrac{1}{4}][C]{butter}\\
		\ingredient[3][Tbs]{flour}\\
		\ingredient[2][C]{milk}\\
		\ingredient[8][oz.]{cream cheese}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[\nicefrac{1}{4}][tsp]{salt}\\
		\ingredient[\nicefrac{1}{4}][tsp]{pepper}\\
		\ingredient[2][tsp]{mustard, Dijon}\\
		\ingredient[2\; \nicefrac{1}{2}][C]{cheddar, sharp}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\ragmarpar{about 6 \ounce~cheese}
\item Preheat oven to $400$ \Fahrenheit, remove cream cheese from fridge to soften, shred cheese.

\item Cook macaroni al dente, drain, set aside.

\item Meanwhile, melt butter over medium heat then stir in flour and cook until bubbling.
	Stir in milk, cream cheese, salt, pepper, and mustard until thick.
	Stir in macaroni and cheese.

\ragmarpar{add as much extra cheese as you want}
\item Add to $\approx9x13"$ or $\approx10x15"$ pan, bake at $400$ \Fahrenheit~for $[20 \ldots 25]$ \minute.
\end{preparation}


\recipeend
