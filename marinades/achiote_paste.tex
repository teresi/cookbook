\section[Achiote Paste]{Achiote Paste~\vegan}\label{achiote_paste_RM}
\index{annato!Achiote~Paste}


\begin{recipestats}[
	servings=\onehalf~$C$,
	preptime=15 \minute,
	source=\citeauthor{rMartinezAchiote} \cite{rMartinezAchiote},
]
\end{recipestats}


\begin{recipeabstract}
	A spice paste often used in marinades or a rub for pork, chicken, and fish.
	Achiote (\textit{achi-o-te}) imparts a strong red color.
	Annato / achiote is sometimes used interchangeably; Annato seeds are grown from the Achiote tree.
\end{recipeabstract}


\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[3][whole]{cloves}\\
		\ingredient[1]{bay leaf}\\
		\ingredient[\onefourth][C]{annato seeds}\\
		\ingredient[2][\teaspoon]{coriander seeds}\\
		\ingredient[2][\teaspoon]{cumin seeds}\\
		\ingredient[2][\teaspoon]{oregano, Mexican}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[1][\teaspoon]{black peppercorns}\\
		\ingredient[1][\teaspoon]{salt}\\
		\ingredient[25][\gram]{garlic}\\
		\ingredient[\onefourth][C]{vinegar, white}\\
		\ingredient[1][\teaspoon]{lime zest}\\
		\ingredient[1][\teaspoon]{orange zest}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Process whole spices in a spice grinder.
	Add to a food processor.

\item Add garlic, vinegar, zest, salt.
	Blend to combine.
	
\item Store in fridge up to 1 $month$.
\end{preparation}
