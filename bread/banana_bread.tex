\section{Banana Bread}


\begin{recipestats}[
	servings=1 loaf,
	preptime=30 \minute,
	bakingtime=1 \hour,
	source=Mike \& Jane,
	original=\citetitle{joyofcooking2006} \cite[p.~628]{joyofcooking2006},
]
\end{recipestats}


\begin{recipeabstract}
	A classic banana bread ripe for customization.
\end{recipeabstract}


\ragmarpar{golden syrup provides a distinct flavor similar to butterscotch or caramel}
\begin{ingredientcolumns}
	\begin{ingredientblock}
		\ingredient[1 \onehalf][C]{flour, all purpose}\\
		\ingredient[\threehalf][\teaspoon]{baking powder}\\
		\ingredient[\onehalf][\teaspoon]{salt}\\
		\ingredient[\twothird][C]{sugar}\\
		\ingredient[6][\Tablespoon]{butter}
	\end{ingredientblock}
	\begin{ingredientblock}
		\ingredient[2]{eggs}\\
		\ingredient[3]{bananas, ripe}\\
		\ingredient[8][\ounce]{strawberries}\\
		\ingredient[\approx 2][\Tablespoon]{golden syrup}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Bring ingredients to room temperature.\\
	Grease a loaf pan.
	Preheat oven to 350 \Fahrenheit.

\item Combine flour, baking powder, salt; set aside.

\ragmarpar{coating w/ flour prevents the fruit from sinking}
\item Quarter strawberries and toss with $\approx1$ \Tablespoon~flour.

\item Beat sugar \& butter on medium until creamy.

\item Beat in eggs, banana.
	Beat in flour $\approx \onehalf$ $C$ at a time.\\
	Fold in strawberries.

\item Pour into loaf pan, bake $\approx1~\hour$.
	Remove from oven when done \& drizzle golden syrup on top.
	Cool, then unmold.
\end{preparation}


\begin{variation}
\item Mike likes strawberries, Jane likes chocolate chips w/o the syrup, the original \cite{joyofcooking2006} uses walnuts and apricot.
\end{variation}


\begin{experiments}
\item Measuring by weight may be more consistent.
\end{experiments}
\recipeend
