\section[Beef Sandwiches]{Peppery Italian Beef Sandwiches}
\index{beef!Beef~Sandwiches}


\begin{recipestats}[
	servings=8,
	preptime=30 \minute,
	bakingtime=10 \hour,
	source=Mike \& Jane,
	original=\citetitle{newCookBook2014} \cite[p.~552]{newCookBook2014},
]
\end{recipestats}


\begin{recipeabstract}
	Chuck roast beef sandwiches with italian peppers.
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[\approx 2 \onehalf][\pound]{beef chuck roast}\\
		\ingredient[2][\teaspoon]{black pepper, ground}\\
		\ingredient[2][\teaspoon]{garlic powder}\\
		\ingredient[14.5][\ounce]{broth (beef $|$ chicken)}\\
		\ingredient[1][\ounce]{dry Italian salad dressing mix}\\
		\ingredient[1][\teaspoon]{onion salt}\\
		\ingredient[1][\teaspoon]{oregano, dry}\\
		\ingredient[1][\teaspoon]{basil, dry}\\
		\ingredient[1][\teaspoon]{parsley, dry}\\
		\ingredient[32][\ounce]{pepperoncini, jarred}\\
		\ingredient[\approx 16][\ounce]{roasted red pepper, jarred}\\
		\ingredient[8][\ounce]{mozzarella}\\
		\ingredient[8]{rolls}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Trim beef, coat with garlic powder \& black pepper.
	Reserve crock pot, coat interior w/ oil.

\item Heat iron skillet on medium high w/ oil and sear the beef.
	Add beef to crockpot.
	Deglaze w/ broth, scrape the fond, add to pot.

\item Drain pepperoncinis, slice off stems, add to pot.

\item Mix remaining spices with broth: dressing mix, onion salt, oregano, basil, parsley.
	Add mixture to pot and submerge peppers.

\item Cook on low for 10 \hour~or high for $[5\dots6]$ \hour.

\ragmarpar{steam heat keeps the bread soft}
\item Heat oven to 325 \Fahrenheit.
	Add cookie sheet on bottom with a little water.
	Add pizza pan on top.

\ragmarpar{also consider a low broil}
\item Slice or shred mozarella.
	Slice rolls.
	Add mozarella to one side of roll and toast in oven $\approx[5\dots10]$ \minute.

\item Meanwhile, remove pepperoncinis, slice red peppers, shred beef.

\item Assemble sandwhich and serve \textit{au jus}.
\end{preparation}


\begin{variation}
\item Serve with pesto mayonnaise (\onehalf~mix of each).
\end{variation}
