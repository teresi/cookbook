\section[Frozen Limeade]{Frozen Limeade~\vegan}


\index{lime!Limeade}


\begin{recipestats}[
	servings=8 people,
	preptime=10 \minute,
	source=Mike \& Jane,
]
\end{recipestats}


\begin{recipeabstract}
	A frozen twist on one of Jane's favorites.
\end{recipeabstract}


\begin{ingredientcolumns}[1]
	\begin{ingredientblock}
		\ingredient[12][\fluidounce]{frozen limeade concentrate}\\
		\ingredient[8][\fluidounce]{coconut juice}\\
		\ingredient[\approx5][C]{ice}\\
		\ingredient[\approx2][dashes]{orange bitters}
	\end{ingredientblock}
\end{ingredientcolumns}


\begin{preparation}
\item Blend coconut juice, ice, and bitters.

\item Blend in limeade concentrate.
	Add ice to your preference.
\end{preparation}


\begin{variation}
\item Serve with maraschino cherries and / or strawberries.
\end{variation}


\begin{experiments}
\item Increase coconut to 17.6 $\fluidounce$?
\end{experiments}

\recipeend
